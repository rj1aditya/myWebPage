C++ Solution


// C++ solution code goes here
#include<bits/stdc++.h>
int majorityElement(vector<int> v) {
	// Write your code here
	map<int,int> m;

	for(int i = 0; i < v.size(); i++)
	{
		m[v[i]]++;
		if(m[v[i]] > v.size()/2)
			return v[i];
	}

	return -1;
}
