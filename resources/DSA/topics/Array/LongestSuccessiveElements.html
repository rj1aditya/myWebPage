//Approach 1
int longestSuccessiveElements1(vector<int>&a) {
    // Write your code here.

    sort(a.begin(), a.end());

    int smallest = INT_MIN;
    int cnt = 1;
    int maxi = 1;
    for(int i = 0; i < a.size(); i++)
    {
        if(a[i]-1 == smallest)
        {
            cnt++;
            maxi = max(maxi, cnt);
            smallest = a[i];
        }
        else if(i > 0 && a[i] == a[i-1])
        {
            continue;
        }
        else{
            smallest = a[i];
            cnt = 1;
        }
    }
    return maxi;
    
}

//Approach 2
int longestSuccessiveElements(vector<int>&a) {
    // Write your code here.

    set<int> s;
    int cnt = 1;
    int maxi = 1;
    for(int i = 0; i < a.size(); i++)
    {
        s.insert(a[i]);
    }

    for(auto x: s)
    {
        //find if no lesser elems are present in the set DS
        //then we will start looking consecutive elems
        if(s.find(x-1) == s.end())
        {
            int elem = x;
            cnt = 1;
            while(s.find(elem+1) != s.end())
            {
                elem = elem+1;
                cnt++;
                maxi = max(maxi, cnt);
            }
        }
    }
    return maxi;
    
}
