long long maxSubarraySum(vector<int> arr, int n)
{
    // Write your code here.

    long long sum = 0;
    long long maxi = LONG_MIN;

    //to print the max sum array elems
    int start = 0;
    int ends = 0;
    int startIdx = 0;
    for(int i = 0; i < n; i++)
    {
        //what is the starting index for max sum
        if(sum == 0) start = i;
        sum += arr[i];
        if(maxi < sum)
        {
            startIdx = start;
            ends = i;
            maxi = sum;
        }
        if(sum < 0)
            sum = 0;
    }
    return maxi>0 ?maxi: 0;
}
