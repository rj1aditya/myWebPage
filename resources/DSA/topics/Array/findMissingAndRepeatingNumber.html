Find the Missing and Repeating Number

C++ Solution


// C++ solution code goes here
vector & lt;
int >
findMissingRepeatingNumbers(vector < int > a)
{
	long long n = a.size();

	// Calculate expected sum and sum of squares for the numbers 1 to n
	long long SN = (n * (n + 1)) / 2; // Sum of first n numbers

	// Sum of squares of first n numbers
	long long S2N = (n * (n + 1) * (2 * n + 1)) / 6;

	// Calculate the actual sum and sum of squares from the array
	long long S = 0;  // Actual sum of the array elements
	long long S2 = 0; // Actual sum of squares of the array elements
	for (int i = 0; i & lt; n; i++)
	{
		S += a[i];
		S2 += (long long)a[i] * (long long)a[i];
	}

	// Calculate the differences between expected and actual values
	long long val1 = S - SN;   // x - y = S - SN
	long long val2 = S2 - S2N; // x^2 - y^2 = S2 - S2N

	// Solve for x and y
	val2 = val2 / val1; // x + y = (S2 - S2N) / (S - SN)

	long long x = (val1 + val2) / 2; // x = (x - y + x + y) / 2
	long long y = x - val1;			 // y = x - (x - y)

	// Return the repeating number first and the missing number second
	return {(int)x, (int)y};
}
