Left Rotate an Array by One

C++ Solution


// C++ solution code goes here
#include <bits/stdc++.h> 
vector<int> rotateArray(vector<int>& arr, int n) {
    // Write your code here.
    
    int bck1 = arr[0];

    for(int i = 1; i < n; i++)
    {
        arr[i-1] = arr[i];
    }
    arr[n-1] = bck1;
    return arr;
}
