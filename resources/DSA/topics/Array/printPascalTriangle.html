Print Pascal’s Triangle

C++ Solution


// C++ solution code goes here
vector<int> genRow(int row)
{
    vector<int> temp;
    temp.push_back(1);
    long long ans = 1;
    for(int col = 1; col < row; col++)
    {
        ans = ans * (row-col);
        ans = ans/col;
        temp.push_back(ans);
    }
    return temp;
}
vector<vector<int>> pascalTriangle(int N) {
    // Write your code here.
    vector<vector<int>> pascalTrin;
    for(int i = 1; i <= N; i++)
    {
        vector<int> temp = genRow(i);
        pascalTrin.push_back(temp);
    }
    return pascalTrin;
}
