//Using hash map
int getLongestSubarray(vector<int> &a, int k) {

    map<long long, int> m;
    int maxi = INT_MIN;
    int sum = 0;
    for(int i = 0; i < a.size(); i++)
    {
        sum += a[i];
        if(sum == k)
        {
            maxi = max(maxi, i+1);
        }

        int leftNumber = sum - k;
        if(m.find(leftNumber) != m.end())
        {
            maxi = max(maxi, i - m[leftNumber]);
        }

        if(m.find(sum) == m.end())
        {
            m[sum] = i;
        }
    }

    return maxi==INT_MIN? 0: maxi;
}
