Rotate array by k step

C++ Solution


// C++ solution code goes here
vector<int> rotateArray1(vector<int>arr, int k) {
    // Write your code here.

    //Brute force
    vector<int> temp;
    int n = arr.size();
    for(int i = 0; i < k; i++)
    {
        temp.push_back(arr[i]);
    }

    for(int i = k; i < n; i++)
    {
        arr[i-k] = arr[i];
    }

    for(int i = n-k; i < n; i++)
    {
        arr[i] = temp[i-(n-k)]; //e.g. 4 - (7 - 3)   
    }

    return arr;
}

//optimal approach
vector<int> rotateArray(vector<int> arr, int k) {
    k = k%arr.size();
    //step1
    //reverse first k part

    reverse(arr.begin(), arr.begin()+k);

    //step2
    //reverse second part
    reverse(arr.begin()+k, arr.end());

    //step3
    //reverse all array
    reverse(arr.begin(), arr.end());

    return arr;
}
