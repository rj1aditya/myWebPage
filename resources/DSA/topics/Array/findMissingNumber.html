int missingNumber1(vector<int>&a, int N) {
    // Write your code here.

    int n = a.size();
    sort(a.begin(), a.end());
    for(int i = 1; i <=n; i++)
    {
        if(a[i-1] != i)
        {
            return a[i-1] - 1;
        }
    }
    return N;
}

//Using XOR
int missingNumber(vector<int> &a, int N) {
    int xor1 = 0;
    int xor2 = 0;

    for(int i = 0; i < a.size(); i++)
    {
        xor1 = xor1^a[i];
        xor2 = xor2^(i+1); //this will give you number from
        //1-a.size()-1
    }
    //take remaining last number 
    xor2 = xor2^N;

    return xor1^xor2;
}
