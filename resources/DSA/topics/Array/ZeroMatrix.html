#include <bits/stdc++.h> 

vector<vector<int>> zeroMatrix1(vector<vector<int>> &matrix, int n, int m) {

    int row[n] = {0}; // row array
    int col[m] = {0}; // col array

    // Traverse the matrix:
    for (int i = 0; i < n; i++) {
        for (int j = 0; j < m; j++) {
            if (matrix[i][j] == 0) {
                // mark ith index of row wih 1:
                row[i] = 1;

                // mark jth index of col wih 1:
                col[j] = 1;
            }
        }
    }

    // Finally, mark all (i, j) as 0
    // if row[i] or col[j] is marked with 1.
    for (int i = 0; i < n; i++) {
        for (int j = 0; j < m; j++) {
            if (row[i] || col[j]) {
                matrix[i][j] = 0;
            }
        }
    }

    return matrix;
}

vector<vector<int>> zeroMatrix(vector<vector<int>> &matrix, int n, int m) {

	int col0 = 1;
	//step 1 mark the row and col to be zero
	for(int i = 0; i < n; i++)
	{
//		if(matrix[i][0] == 0)
//			col0 = 0;
		for(int j = 0; j < m; j++)
		{
			if(matrix[i][j] == 0)
			{
				matrix[i][0] = 0;

				if(j != 0)
					matrix[0][j] = 0;
				else
					col0 = 0;
			} 

		}
	}

	//step 2
	for(int i = 1; i < n; i++)
	{
		for(int j = 1; j < m; j++)
		{
			if((matrix[i][0] == 0 || matrix[0][j] == 0))
				matrix[i][j] = 0;
		}
	}

	//check 1st row
	if(matrix[0][0] == 0)
		for(int i = 0; i < m; i++)
			matrix[0][i] = 0;

	//check 1st col
	if(col0 == 0)
		for(int i = 0; i < n; i++)
			matrix[i][0] = 0;


	return matrix;
}
