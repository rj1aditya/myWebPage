int getSingleElement(vector<int> &arr) {

	int l = 0;
	int r = arr.size()-1;

	while(l <= r)
	{
		int m = (l+r)/2;

		if(arr[m] != arr[m-1] && arr[m] != arr[m+1])
			return arr[m];
		//below we are check the right part of the arr
		else if(m%2 == 1 && arr[m] == arr[m+1] || m%2==0 && arr[m] == arr[m-1])
		{
			r = m-1;
		}
		//here it will consider the left part of the array
		else
		{
			l = m+1;
		}
	}
	return -1;
}
