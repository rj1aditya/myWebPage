Count All Subarrays With Given Sum

C++ Solution


// C++ solution code goes here
int findAllSubarraysWithGivenSum(vector < int > & arr, int k) {
    // Write Your Code Here

    //<sum, count>
    map<long long, int> m;
    m[0] = 1;
    int cnt = 0;
    int sum=0;
    for(int i = 0; i < arr.size(); i++)
    {
        sum += arr[i];

        if(m.find(sum-k) != m.end())
        {
            cnt += m[sum-k];
        }
        m[sum]++;
    }
    return cnt;
}
