Subarrays with XOR ‘K’

C++ Solution


// C++ solution code goes here
#include<bits/stdc++.h>
int subarraysWithSumK(vector < int > a, int b) {
    // Write your code here

    map<int, int> m;
    m[0] = 1;

    int cnt = 0;
    int xr = 0;
    for(int i = 0; i < a.size(); i++)
    {
        xr = xr^a[i];

        //formula
        int x = xr^b;
        if(m.find(x) != m.end())
        {
            cnt += m[x];
        }
        m[xr]++;
    }
    return cnt;
    
}
