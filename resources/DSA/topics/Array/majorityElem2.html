Majority Element II

C++ Solution


// C++ solution code goes here
#include <vector>
#include <algorithm>
using namespace std;

vector<int> majorityElement(vector<int> v) {
    // Initialize variables
    int el1 = INT_MAX, el2 = INT_MAX;  // Use a large sentinel value
    int cnt1 = 0, cnt2 = 0;

    // Moore's Voting Algorithm to find the two most frequent candidates
    for(int i = 0; i < v.size(); i++) {
        if(cnt1 == 0 && v[i] != el2) {
            el1 = v[i];
            cnt1 = 1;
        } else if(cnt2 == 0 && v[i] != el1) {
            el2 = v[i];
            cnt2 = 1;
        } else if(el1 == v[i]) {
            cnt1++;
        } else if(el2 == v[i]) {
            cnt2++;
        } else {
            cnt1--;
            cnt2--;
        }
    }

    // Reset counts for final validation
    cnt1 = cnt2 = 0;
    for(int i = 0; i < v.size(); i++) {
        if(el1 == v[i]) cnt1++;
        else if(el2 == v[i]) cnt2++;
    }

    // Prepare the result
    vector<int> ans;
    if(cnt1 > v.size() / 3) ans.push_back(el1);
    if(cnt2 > v.size() / 3) ans.push_back(el2);

    // Return the result (sorted if needed)
    sort(ans.begin(), ans.end());
    return ans;
}
