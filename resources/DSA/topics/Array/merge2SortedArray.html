vector < int > sortedArray(vector < int > a, vector < int > b) {
    // Write your code here

    //in brute force we can solve this problem
    //using set DS easily

    //optimal approach using 2 pointer
    int i = 0;
    int j = 0;
    int n1 = a.size();
    int n2 = b.size();

    vector<int> ans;
    while(i < n1 && j < n2)
    {
        if(a[i] <= b[j])
        {
            if(ans.size() == 0 || a[i] != ans.back()) ans.push_back(a[i]);
            i++;
        }
        else
        {
            if(ans.size() == 0 || b[j] != ans.back() ) ans.push_back(b[j]);
            j++;
        }
        /*else if(a[i] == b[j])
        {
            if(ans.size() == 0 || b[j] != ans.back()) ans.push_back(a[i]);
            i++;
            j++;
        }*/
    }
    while(i < n1)
    {
        if(a[i] != ans.back())
        {
            ans.push_back(a[i]);
        }
        i++;
    }
    while(j < n2)
    {
        if(b[j] != ans.back())
        {
            ans.push_back(b[j]);
        }
        j++;
    }
    return ans;
}
