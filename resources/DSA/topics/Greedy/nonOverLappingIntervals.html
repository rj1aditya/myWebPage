Non Overlapping Intervals

// C++ solution code goes here
bool compa(const vector<int> &p1, const vector<int> &p2)
{
    return p1[1] < p2[1];
}


class Solution {
public:
    int eraseOverlapIntervals(vector<vector<int>>& intervals) {
        sort(intervals.begin(), intervals.end(), compa);

        int maxChain = 0;

        int prevEndTimeInd = 0;
        maxChain = 1;
        for(int i = 1; i < intervals.size(); i++)
        {
            if(intervals[i][0] >= intervals[prevEndTimeInd][1])
            {
                maxChain++;
                prevEndTimeInd = i;
            }
        }

        return intervals.size() - maxChain;
    }
};
