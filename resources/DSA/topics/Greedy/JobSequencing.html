Job Sequencing Problem

C++ Solution


// C++ solution code goes here
#include <vector>
#include <algorithm>
using namespace std;

struct Job
{
    int id, deadline, profit;
};

// Comparator function to sort jobs by profit in descending order
bool compa(const vector<int> &j1, const vector<int> &j2)
{
    return j1[2] > j2[2]; // Sort by profit (j[2]) in descending order
}

vector<int> jobScheduling(vector<vector<int>> &jobs)
{
    // Sort jobs by profit in descending order
    sort(jobs.begin(), jobs.end(), compa);

    // Find the maximum deadline to set up our scheduling array
    int maximumDeadline = -1;
    for (int i = 0; i < jobs.size(); i++)
    {
        maximumDeadline = max(maximumDeadline, jobs[i][1]);
    }

    // Initialize the slot array with -1 (unoccupied)
    vector<int> hash(maximumDeadline + 1, -1);

    int jobsCnt = 0;
    int maxProfit = 0;

    // Iterate over each job
    for (int i = 0; i < jobs.size(); i++)
    {
        // Try to find a free slot for this job, starting from its deadline
        for (int j = jobs[i][1]; j > 0; j--)//job index started from 1 not 0 so j>0 taken
        {
            if (hash[j] == -1) // If the slot is free
            {
                hash[j] = i; // Assign this slot
                maxProfit += jobs[i][2]; // Add the profit
                jobsCnt++; // Increment job count
                break; // Move to the next job
            }
        }
    }

    // Return the total number of jobs done and the total profit
    return {jobsCnt, maxProfit};
}
