#include <bits/stdc++.h>
int partition(vector<int> &arr, int s, int e)
{
    int pivot = arr[e];
    int i = s;

    for(int c = s; c < e; c++)
    {
        if(arr[c] < pivot)
        {
            swap(arr[i], arr[c]);
            i++;
        }
    }
    swap(arr[i], arr[e]);
    return i;
}

void qSort(vector<int> &arr, int s, int e)
{
    if(s<e)
    {
        int pi = partition(arr, s, e);
        qSort(arr, s, pi-1);
        qSort(arr, pi+1, e);
    }
}

vector<int> quickSort(vector<int> arr)
{
    // Write your code here.
    qSort(arr, 0, arr.size()-1);
    return arr;
}
