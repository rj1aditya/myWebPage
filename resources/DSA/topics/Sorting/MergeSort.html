#include <iostream>
#include <vector>

using namespace std;
class Solution
{
public:
    void merge(int arr[], int l, int m, int h)
    {
        // Your code here
        vector<int> ds;
        int left = l;
        int right = m + 1;

        while (left <= m && right <= h)
        {
            if (arr[left] < arr[right])
                ds.push_back(arr[left++]);
            else
                ds.push_back(arr[right++]);
        }
        while (left <= m)
        {
            ds.push_back(arr[left++]);
        }
        while (right <= h)
        {
            ds.push_back(arr[right++]);
        }

        for (int i = l; i <= h; i++)
            arr[i] = ds[i - l];
    }

public:
    void mergeSort(int arr[], int l, int h)
    {
        // code here
        //  if(l==h)
        //      return;
        if (l < h)
        {
            int mid = (l + h) / 2;
            mergeSort(arr, l, mid);
            mergeSort(arr, mid + 1, h);
            merge(arr, l, mid, h);
        }
    }
};

int main()
{
    Solution s;
    int arr[] = {5, 1, 0, 4, 6, 3, 2};
    s.mergeSort(arr, 0, 6);

    for (int i = 0; i <= 6; i++)
        cout << arr[i] << " ";
}
