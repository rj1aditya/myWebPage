// C++ solution code goes here
class Solution{
	public:
		vector<string> AllPossibleStrings(string str){
		    // Code here
		    
		    int size = str.length();
		    vector<string> ans;
		    //loop will run through 2^size -1
		    for(int num = 0; num < (1 << size); num++)
		    {
		        //check which bit is set in a num
		        string s = "";
                for(int n = 0; n < size; n++)
                {
                    int x = 1;
                    x = x << n;
                    if(x & num)
                    {
                        s = s+ str[n];
                    }
                }
                if(s.length() > 0)
                    ans.push_back(s);
		    }
		    sort(ans.begin(), ans.end());
		    return ans;
		}
};
