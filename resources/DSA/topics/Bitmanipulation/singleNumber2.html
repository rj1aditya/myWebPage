class Solution {
    public:
        int singleNumber1(vector<int>& nums) {
            //will check all the bits in the number from 0 to 31
            //we check same bit in all the number
            //then move to another bit
            int ans = 0;
            for(int bitIndex = 0; bitIndex < 32; bitIndex++)
            {
                int count = 0;
                //here we check the bit in all the number
                //then count the number of that bit set in which number
                for(int i = 0; i < nums.size(); i++)
                {
                    if(nums[i] & ( 1 << bitIndex ) )
                        count++;
                }
                //if number is more than 3 times and not multiple of 3 
                //it means this bit is part of that unique number
                if(count % 3 == 1)
                    ans = ans | ( 1<<bitIndex );
            }
            return ans;
        }
        
        int singleNumber2(vector<int>& nums) {
    
            sort(nums.begin(), nums.end());
            for(int i = 1; i < nums.size(); i = i+3)
            {
                if(nums[i] != nums[i-1])
                    return nums[i-1];
            }
    
            return nums[nums.size()-1];
        }
    
        int singleNumber(vector<int>& nums) {
            int ones = 0;
            int twos = 0;
    
            for(int i = 0; i < nums.size(); i++)
            {
                //It means if number is not in two so ADD in ones, if
                //Number in already in ones then delete this number from One,
                //later statement will add this number
                ones = ones ^ nums[i] & (~twos);

                //It means if number is in ones so ADD in two
                twos = twos ^ nums[i] & (~ones);
            }
            return ones;
        }
    };
