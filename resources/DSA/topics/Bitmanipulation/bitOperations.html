void swap(int &x, int &y) {
    cout << "Before swap value of x and y are: " << x << " " << y << endl;

    // XOR swap logic
    x = x ^ y; // Step 1: x now holds (x XOR y)
    y = x ^ y; // Step 2: y now holds the original value of x
    x = x ^ y; // Step 3: x now holds the original value of y

    cout << "After swap value of x and y are: " << x << " " << y << endl;
}


// Function to check if the nth bit is set
bool checkNthBit(int num, int nth) {
    int x = 1;        // Start with 1 (000...001 in binary)
    x = x << nth;     // Shift 1 left by `nth` positions (to create a mask)

    // Perform bitwise AND to isolate the nth bit and return true if it's set.
    return (num & x); 
}


// Function to check if the nth bit is set using right shift operator
bool checkNthBit(int num, int nth) {
    num = num >> nth;  // Right-shift the number by `nth` positions
    return (num & 1);  // Check the least significant bit (LSB)
}

// Function to set the nth bit
void setNthBit(int &num, int nth) {
    int x = 1;        // Start with 1 (000...001 in binary)
    x = x << nth;     // Shift 1 left by `nth` positions to create a mask

    num = num | x;    // Use bitwise OR to set the nth bit
}

// Function to clear the nth bit
void clearNthBit(int &num, int nth) {
    int x = 1;         // Start with 1 (000...001 in binary)
    x = x << nth;      // Shift 1 left by `nth` positions to create a mask
    x = ~x;            // Flip all bits to create a mask where the nth bit is 0 and all others are 1

    num = num & x;     // Use bitwise AND to clear the nth bit
}

// Function to toggle the nth bit
void toggleNthBit(int &num, int nth) {
    int x = 1;         // Start with 1 (000...001 in binary)
    x = x << nth;      // Shift 1 left by `nth` positions to create a mask

    num = num ^ x;     // Use bitwise XOR to toggle the nth bit
}

// Function to remove the last set bit
void removeLastSetBit(int &num) {
    num = num & (num - 1); // Clears the last set bit
}

bool checkNumberIsPowerOf2(int num)
{
    // A number is a power of 2 if:
    // 1. It is greater than 0.
    // 2. It has only one set bit, which is true when `num & (num - 1) == 0`.
    return num > 0 && (num & (num - 1)) == 0;
}
