Minimum Bit Flips to Convert Number

C++ Solution


// C++ solution code goes here
class Solution {
    public:
        int minBitFlips(int start, int goal) {
            
            //it will set and flip all those bit which needed 
            //to be flip to make goal from start
            int ans = start ^ goal;
    
            //count number of bit set in ans
            int count = 0;
            while(ans)
            {
                ans = ans & (ans-1);
                count++;
            }
            return count;
        }
    };
