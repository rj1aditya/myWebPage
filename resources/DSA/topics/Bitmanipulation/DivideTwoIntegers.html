Divide Two Integers

class Solution {
    public:
        int divide(int dividend, int divisor) {
            // Handle overflow case
            if (dividend == INT_MIN && divisor == -1) return INT_MAX;
    
            // Determine the sign of the result
            bool sign = (dividend < 0) == (divisor < 0);
    
            // Convert to positive values using long to avoid overflow
            long N = abs((long)dividend);
            long d = abs((long)divisor);
    
            long quotient = 0;
    
            // Perform bitwise division
            while (N >= d) {
                long cnt = 0;
                while (N >= (d << (cnt + 1))) {
                    cnt++;
                }
                quotient += (1L << cnt);  // Add 2^cnt to quotient
                N -= (d << cnt);         // Subtract d * 2^cnt from N
            }
    
            // Apply sign and clamp the result
            quotient = sign ? quotient : -quotient;
            if (quotient > INT_MAX) return INT_MAX;
            if (quotient < INT_MIN) return INT_MIN;
    
            return quotient;
        }
    };
