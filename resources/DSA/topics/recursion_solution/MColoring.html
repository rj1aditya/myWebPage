M-Coloring Problem

C++ Solution


// C++ solution code goes here
class Solution{
public:
    bool isColor(int node, bool graph[101][101], vector<int> &color, int col)
    {
        for(int k = 0; k <= color.size(); k++)
        {
            if(k != node && graph[k][node] == 1 && color[k] == col)
                return false;
        }
        return true;
    }
    
    bool gColor(bool graph[101][101], int m, vector<int> &color, int node)
    {
        if(node == color.size())
            return true;
            
        for(int i = 1; i <= m; i++)
        {
            if(isColor(node, graph, color, i))
            {
                color[node] = i;
                if(gColor(graph, m, color, node+1) == true)
                    return true;
                    
                color[node] = -1;
            }
        }
        return false;
    }
    // Function to determine if graph can be coloured with at most M colours such
    // that no two adjacent vertices of graph are coloured with same colour.
    bool graphColoring(bool graph[101][101], int m, int n) {
        // your code here
        vector<int> color(n, -1);
        if(gColor(graph, m, color, 0) == true)
            return true;
            
        return false;
    }
};
