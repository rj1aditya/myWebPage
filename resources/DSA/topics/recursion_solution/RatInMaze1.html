class Solution{
    vector<int> di{1,0,0,-1};
    vector<int> dj{0,-1,1,0};
    
    public:
    void findP(vector<vector<int>> &m, int n, vector<string> &ans, int i, int j, string move, vector<vector<int>> &visited)
    {
        if(i == n-1 && j == n-1)
        {
            ans.push_back(move);
            return;
        }
        string dir = "DLRU";
        
        for(int k = 0; k < 4; k++)
        {
            int inextMove = i + di[k];
            int jnextMove = j + dj[k];
            
            if(inextMove >= 0 && inextMove < n && jnextMove >= 0 && jnextMove < n
                && visited[inextMove][jnextMove] == 0 && m[inextMove][jnextMove] == 1)
                {
                    visited[i][j] = 1;
                    findP(m, n, ans, inextMove, jnextMove, move+dir[k], visited);
                    visited[i][j] = 0;
                }
        }
    }
    vector<string> findPath(vector<vector<int>> &m, int n) {
        // Your code goes here
        vector<string> ans;
        vector<vector<int>> visited(n, vector<int>(n, 0));        
        
        if(m[0][0]== 1)
            findP(m, n, ans, 0, 0, "", visited);

        return ans;
    }
};
