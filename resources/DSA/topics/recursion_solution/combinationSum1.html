void fun(vector& candidates, vector& ds, int target, int idx, int sum)
{
    if(sum == target)
    {
        ans.push_back(ds);
        return;
    }
    if(idx == candidates.size())
    {
        return;
    }
    if(sum+candidates[idx] <= target)
    {
        ds.push_back(candidates[idx]);
        fun(candidates, ds, target, idx, sum+candidates[idx]);
        ds.pop_back();
    }
    fun(candidates, ds, target, idx+1, sum);

}
