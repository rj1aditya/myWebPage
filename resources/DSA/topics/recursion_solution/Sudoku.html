class Solution
{
    private boolean isValid(int row, int col, char [][]b, char var)
    {
        for(int i = 0; i < b.length; i++)
        {
            if(b[i][col] == var)
                return false;
            if(b[row][i] == var)
                return false;

            if( b[3*(row/3) + i/3][3*(col/3) + i%3] == var )
                return false;
        }
        return true;
    }
    private  boolean fun(char [][]b)
    {
        for(int i = 0; i < b.length; i++)
        {
            for(int j = 0; j < b[0].length; j++)
            {
                if(b[i][j] == '.')
                {                
                    for(char val = '1'; val <= '9'; val++)
                    {
                        if(isValid(i, j, b, val))
                        {
                            b[i][j] = val;
                            if(fun(b) == true)
                                return true;
                            else
                                b[i][j] = '.'; 
                        }
                        
                    }
                    return false;
                }
            }
        }
        return true;
    }

    public void solveSudoku(char[][] board) 
    {
        fun(board);
    }
}
