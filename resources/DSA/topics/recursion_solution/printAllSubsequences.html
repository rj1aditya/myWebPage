Print All Subsequences of an array or string

C++ Solution


// C++ solution code goes here
class solution
{
    public:
    void printAllSubsequences(int arr[], vector &ds, int i, int n)
    {
        if( i >= n)
        {
            if(ds.size() > 0)
            {
                for(int k = 0; k < ds.size(); k++)
                    cout << ds[k] << " ";
            }
            else
                cout << "{}";
            
            cout << endl;
            return;
        }

        printAllSubsequences(arr, ds, i+1, n);
        ds.push_back(arr[i]);
        printAllSubsequences(arr, ds, i+1, n);
        ds.pop_back();
    }
};

import java.util.*;
                        
public class printAllSubSequences {
    public static void printSubSeq(int[] arr, ArrayList ds, int indx)
    {
        if(indx >= arr.length)
        {
            if(ds.size() > 0)
            {
                System.out.print(ds);
            }
            else
                System.out.print("{}");

                System.out.println();

            return;
        }
        ds.add(arr[indx]);
        printSubSeq(arr, ds, indx+1);
        //in vector ds.pop_back() used while here remove used to remove last indx var
        ds.remove(ds.size()-1);
        printSubSeq(arr, ds, indx+1);
    }
            
    public static void main(String[] args)
    {
        ArrayList ds = new ArrayList<>();
        int[] arr = {1,2,3};
        printSubSeq(arr, ds, 0);
    }
}
