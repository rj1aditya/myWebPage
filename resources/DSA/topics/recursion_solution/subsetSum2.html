class Solution {
  vector<vector<int>> ans;
  vector<int> dp;
  
  void f(vector<int>& arr, int ind)
  {
      ans.push_back(dp);
      
      for(int i = ind; i < arr.size(); i++)
      {
          if(i != ind && arr[i] == arr[i-1]) continue;
          
          dp.push_back(arr[i]);
          f(arr, i+1);
          dp.pop_back();
      }
  }
public:
  vector<vector<int>> subsetsWithDup(vector<int>& nums) {
      
      sort(nums.begin(), nums.end());
      f(nums, 0);
      return ans;
  }
};
