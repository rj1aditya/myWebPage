class Solution
{
    bool isSubSetSum(vector<int>& arr, int sum, vector<vector<int>>& dp, int ind)
    {
        if(ind == arr.size())
            return false;

        if (sum == 0 || sum == arr[ind])
            return true;

        if (dp[ind][sum] != -1)
            return dp[ind][sum];

        if(sum >= arr[ind])
            if (isSubSetSum(arr, sum-arr[ind], dp, ind + 1) == true)
                return true;

        if (isSubSetSum(arr, sum, dp, ind+1) == true)
            return true;
        
        return dp[ind][sum] = false;
    }
public:
    bool isSubsetSum(vector<int>arr, int sum){
        // code here
        vector<vector<int>> dp(arr.size(), vector<int>(sum+1, -1) );
        return isSubSetSum(arr, sum, dp, 0);
    }
};
