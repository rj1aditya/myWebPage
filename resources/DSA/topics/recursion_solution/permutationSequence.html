#include<string>
class Solution {

    void findAns(vector<int> &arr, int &factor, int &k, string &ans)
    {
        if(arr.size() == 0)
            return;
        factor = factor/arr.size();
        ans = ans + to_string(arr[k/factor]);
        arr.erase(arr.begin() + k/factor);
        k = k % factor;
        findAns(arr, factor, k, ans);
    }
public:
    string getPermutation(int n, int k) {
        int factor = 1;
        vector<int> arr;
        for(int i = 1; i <= n; i++){
            factor *= i;
            arr.push_back(i);
        }  
        //arr.push_back(n);
        k = k-1;
        string ans="";
        findAns(arr, factor, k, ans);
        return ans;
    }
};
