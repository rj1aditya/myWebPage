            using namespace std;
            
            // Print 1 to N using backtracking
            void print1ToN(int n) {
                if (n > 0) {
                    print1ToN(n - 1);
                    cout << n << endl;
                }
            }
            
            // Print N to 1 using backtracking
            void printNTo1(int i, int n) {
                if (i > n)
                    return;
            
                printNTo1(i + 1, n);
                cout << i << endl;
            }
            
            int main() {
                int n = 5;
                print1ToN(n);
                printNTo1(1, n);
                return 0;
            }
