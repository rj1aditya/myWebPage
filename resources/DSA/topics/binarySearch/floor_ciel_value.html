int findCiel(vector<int> arr, int n, int x) {
	// Write your code here

	int l = 0, r = n-1;
	int idx=n;
	while(l <= r)
	{
		int mid = l + (r-l)/2;

		if(arr[mid]>= x)
		{
			idx = mid;
			r = mid-1;
		}
		else
			l = mid+1;
	}

	if(idx != n)
		return arr[idx];
	return -1;
}


int findFloor(vector<int> arr, int n, int x) {
	// Write your code here

	int l = 0, r = n-1;
	int idx=n;
	while(l <= r)
	{
		int mid = l + (r-l)/2;

		if(arr[mid] <= x)
		{
			idx = mid;
			l = mid+1;
		}
		else
			r = mid-1;
	}
	if(idx != n)
		return arr[idx];
	return -1;
}


pair<int, int> getFloorAndCeil(vector<int> &a, int n, int x) {
	// Write your code here.

	int ceil = findCiel(a, n, x);
	int floor = findFloor(a, n, x);


	return {floor, ceil};

}
