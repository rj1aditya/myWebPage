int findMaxElemInMidCol(vector<vector<int>> &g, int midCol)
	{
		int rows = g.size();
		int maxi = -1e9-1;
		int indx=-1;
		for(int i = 0; i < rows; i++)
		{
			if(maxi < g[i][midCol])
			{
				maxi = g[i][midCol];
				indx = i;
			}
		}
		return indx;
	}
	
	vector<int> findPeakGrid(vector<vector<int>> &g){
		// Write your code here. 
	
		int n = g.size();
		int m = g[0].size();
	
		int low = 0, high = m-1;
	
		while(low <= high)
		{
			int midCol = (low+high)/2;
	
			int rowOfMaxElemIdx = findMaxElemInMidCol(g, midCol);
			int maxValElem = g[rowOfMaxElemIdx][midCol];
	
			int left = midCol-1;
			int right = midCol+1;
	
			int leftVal = left >= 0 ? g[rowOfMaxElemIdx][left]: -1;
			int rightVal = right < m ? g[rowOfMaxElemIdx][right]: -1;
			if(maxValElem > leftVal && maxValElem > rightVal)
				return {rowOfMaxElemIdx, midCol};
			else if(maxValElem < leftVal)
				high = midCol-1;
			else if(maxValElem < rightVal)
				low = midCol+1;
		}   
		return {-1,-1};
	}
