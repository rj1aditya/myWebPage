int findNthRoot(int mid, int n, int num)
{
  long long x = 1;
  for(int i = 1; i <= n; i++)
  {
    x = mid*x;
    if(x > num)
      return 2;
  }

  if(x == num)
    return 1;
  
  return 0;
}

int NthRoot(int n, int m) {
  // Write your code here.

  if(m == 0 || m == 1)
    return m;

  int l = 1, r = m, ans = 1;
  while(l <= r)
  {
    int mid = l + (r-l)/2;

    int x = findNthRoot(mid, n, m);
    if(x == 1)
      return mid;
    else if(x == 0)
      l = mid+1;
    else
      r = mid-1;
  }
  return -1;
}
