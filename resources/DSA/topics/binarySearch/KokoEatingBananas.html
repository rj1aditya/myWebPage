// C++ solution code goes here
#include <iostream>
	#include <vector>
	#include <cmath>
	#include <algorithm> // For max function
	
	using namespace std;
	
	long long findTotalHr(long long hr, vector<int> &v) {
		int n = v.size();
		long long ans = 0;
		for(int i = 0; i < n; i++) {
			// Use ceil to calculate the time required for each pile
			ans += (v[i] + hr - 1) / hr; // More efficient way to 
            //calculate ceil(v[i] / hr)
			
			// Early exit if the time already exceeds available hours
			if(ans > 1000000000) return ans;
		}
		return ans;
	}
	
	int findMax(vector<int> &v) {
		int maxi = v[0];
		for(int i = 1; i < v.size(); i++) // Start loop from 1
			maxi = max(maxi, v[i]);
		return maxi;
	}
	
	int minimumRateToEatBananas(vector<int> v, int h) {
		// Edge case: if the vector is empty
		if(v.empty()) return -1;
	
		int n = v.size();
		// Edge case: if h < n, it's impossible to eat all 
        //bananas
		if(h < n) return -1;
	
		// Range will be from 1 banana/hr to the maxBanana 
        //in the array
		long long l = 1, r = findMax(v);
		long long m = 0;
	
		while(l <= r) {
			m = l + (r - l) / 2;
	
			// m is the banana eating rate per hour
			long long timeReq = findTotalHr(m, v);
	
			if(timeReq <= h) {
				r = m - 1;
			} else {
				l = m + 1;
			}
		}
		return l; // l will be the minimum rate to 
        //eat all bananas in h hours
	}
