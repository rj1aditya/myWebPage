bool isPossibleToPlace(vector<int> &stalls, int cows, int minOfMax)
	{
		int cow = 1;
		int lastPlacedCow = stalls[0];
	
		for(int i = 1; i < stalls.size(); i++)
		{
			if(stalls[i] - lastPlacedCow >= minOfMax)
			{
				lastPlacedCow = stalls[i];
				cow++;
				if(cow == cows)
					return true;
			}
		}
		return false;
	}
	
	int aggressiveCows(vector<int> &stalls, int k)
	{
		//Write your code here.
		sort(stalls.begin(), stalls.end());
		int n = stalls.size();
		int l = 1;
		int r = stalls[n-1] - stalls[0];
		int ans = -1;
		while(l <= r)
		{
			int mid = l + (r-l)/2;
	
			if (isPossibleToPlace(stalls, k, mid) == true) {
				ans = mid;
				l = mid + 1;
			} 
			else
				r = mid-1;
		}
		return r;
	}
