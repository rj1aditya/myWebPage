K-th element of two sorted arrays

C++ Solution


// C++ solution code goes here
#include<bits/stdc++.h>
int kthElement(vector<int> &arr1, vector<int>& arr2, int n, int m, int k){
    // Write your code here

    if(n > m)
        return kthElement(arr2, arr1, m, n, k);

    //here we start low from max(k-m , 0), coz suppose k is very big,
    //e.g. k = 10, and n = 6, m = 7, then low = k-m, 10-7 = 3, atleast 3 elems
    //needs to pick from first array, if we start 0 as low, then 2nd array won't
    //able to make all 10 elems from 2nd array coz it has total 7 elems in it.
    int low = max(k-m, 0);

    
    int high = min(k, n);

    int left = k;

    while(low <= high)
    {
        int mid1 = (low+high)/2;
        int mid2 = left-mid1;

        int r1 = mid1 < n ? arr1[mid1]: INT_MAX;
        int r2 = mid2 < m ? arr2[mid2]: INT_MAX;
        int l1 = mid1 > 0 ? arr1[mid1-1]: INT_MIN;
        int l2 = mid2 > 0 ? arr2[mid2-1]: INT_MIN;

        if(l1 <= r2 && l2 <= r1)
            return max(l1, l2);
        else if(l1 > r2) high = mid1-1;
        else low = mid1+1;  
    }
    return 0;
}
