int findPeakElement(vector<int> &arr) {
    // Write your code here

    int n = arr.size();
    if(n == 1)
        return 0;
    
    int l = 0, r = n-1;

    //Initial test case if find peak at the 0th index or last index itself,
    //just return those only
    if(arr[l] > arr[l+1])
        return l;
    if(arr[r] > arr[r-1])
        return r;
    
    l++;
    r--;

    while(l <= r)
    {
        int m = l + (r-l)/2;

        //Check peak elem
        if(arr[m] > arr[m-1] && arr[m] > arr[m+1])
            return m;
        else if( arr[m] > arr[m-1]) //here left part elminate, incr. slope
        {
            l = m+1;
        }
        else if( arr[m] > arr[m+1]) //here right part eliminate, decr. slope
        {
            r = m-1;
        }
        else
            l = m+1;
    }
    return -1;
}
