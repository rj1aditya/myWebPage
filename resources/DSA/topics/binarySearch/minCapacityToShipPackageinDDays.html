Capacity To Ship Packages Within D Days

int findleastWeightCapacity(int minweight, vector<int> &weights, int d)
	{
		int days = 1;
		int sum = 0;
		for(int i = 0; i < weights.size(); i++)
		{
			if(sum+weights[i] <= minweight)
			{
				sum = sum+weights[i];
			}        
			else
			{
				days++;
				sum = weights[i];
			}
		}
		return days <= d;
	}
	
	int leastWeightCapacity(vector<int> &weights, int d)
	{
		// Write your code here.
	
		if(weights.empty()) return -1;
		//minweight is the max weight available in the package
		//maxweight is the sum of the total item available in the package
		int l=0, r=0;
		int n = weights.size();
	
		for(int i = 0; i < n; i++)
		{
			l = max(weights[i], l);
			r += weights[i];
		}
	
		while(l <= r)
		{
			int minweight = l + (r-l)/2;
	
			int isPossible = findleastWeightCapacity(minweight, weights, d);
	
			if(isPossible)
			{
				r = minweight-1;
			}
			else
				l = minweight+1;
		}
		return l;
	}
