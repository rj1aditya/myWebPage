int findBookAllocation(int minpages,vector<int> &arr, int totalBooks, int totalStudent)
	{
		int student = 1;
		int pages = 0;
	
		for(int i = 0; i < totalBooks; i++)
		{
			if(arr[i] + pages <= minpages)
			{
				pages += arr[i];
			}
			else
			{
				student++;
				pages = arr[i];
			}
		}
	
		return student;
	}
	int findPages(vector<int>& arr, int n, int m) {
		// Write your code here.
		if(m > n) return -1;
		int low = *max_element(arr.begin(), arr.end());
		int high = accumulate(arr.begin(), arr.end(), 0);
	
		while(low <= high)
		{
			int minpages = low + (high - low)/2;
	
			int pagesAllocatedToStudent = findBookAllocation(minpages, arr, n, m);
	
			if(pagesAllocatedToStudent > m)
			{
				low = minpages + 1;
			}
			else
				high = minpages-1;
		}
	
		return low;
	}
