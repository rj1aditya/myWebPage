int floorSqrt1(int n)
{
    // Write your code here.
    if(n == 0 || n == 1)
        return n;
    int ans = 1;
    for(int i = 1; i < n; i++)
    {
        if(i*i <= n)
            ans = i;
        else
            return ans;
    }

    return -1;
}

//Using Binary search algo
int floorSqrt(int n)
{
    // Write your code here.
    if(n == 0 || n == 1)
        return n;
    int l = 1, r = n, ans = 1;
    while(l <= r)
    {
        long long mid = l + (r-l)/2;

        if(mid*mid == n)
            return mid;

        if(mid*mid <= n)
        {
            ans = mid;
            l = mid+1;
        }
        else
            r = mid-1;
    }
    return ans;
}
