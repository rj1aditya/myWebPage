pair<int,int> findMinAndMax(vector<int> &arr)
	{
		int maxi = arr[0];
		int mini = arr[0];
	
		for(int i = 1; i < arr.size(); i++)
		{
			maxi = max(maxi, arr[i]);
			mini = min(mini, arr[i]);
		}
	
		return {mini, maxi};
	}
	int isPossibleToMakeBuequet(vector<int> arr, int mid, int flow, int bue)
	{
		int cntflower=0;
		int cntbue=0;
		for(int i = 0; i < arr.size(); i++)
		{
			if(arr[i] <= mid)
			{
				cntflower++;
			}
			else
			{
				cntbue += (cntflower/flow);
				cntflower=0;
			}
		}
		cntbue += (cntflower/flow);
		return cntbue >= bue;
	}
	
	int roseGarden(vector<int> arr, int k, int m)
	{
		// Write your code here
	
		if(arr.empty()) return -1;
	
		auto p = findMinAndMax(arr);
		int maxi = p.second;
		int mini = p.first;
	
		int n = arr.size();
	
		//k*m is the total flower needed to make buquete
		//n is the total flower available in the arr vector
		if( k*m > n)
			return -1;
	
		while(mini <= maxi)
		{
			int mid = mini + (maxi-mini)/2;
	
			int x = isPossibleToMakeBuequet(arr, mid, k, m);
			if(x == 1)
			{
				maxi = mid-1;
			}
			else
				mini = mid+1;
		}
		return mini;
	}
