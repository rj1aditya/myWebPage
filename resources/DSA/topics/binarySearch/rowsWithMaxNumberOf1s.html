int lower_bound(vector<int> v, int m, int x)
	{
		int low = 0, high = v.size()-1;
		int ans = m;
	
		while(low <= high)
		{
			int mid = (low+high)/2;
			if(v[mid] == x)
			{
				ans = mid;
				high = mid-1;
			}
			else
				low = mid+1;
		}
		//cout << "ans:"<<ans<<endl;
		return ans;
	}
	
	int rowWithMax1s(vector<vector<int>> &matrix, int n, int m) {
		// Write Your Code Here
		int cnt_max=0;
		int idx = -1;
		for(int i = 0; i < n; i++)
		{
			int cnt_ones = m-lower_bound(matrix[i], m, 1);
			if(cnt_ones > cnt_max)
			{
				cnt_max = cnt_ones;
				idx = i;
			}
		}
		return idx;
	}
