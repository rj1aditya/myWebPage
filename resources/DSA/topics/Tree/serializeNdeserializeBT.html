Serialize and De-serialize Binary Tree

C++ Solution

/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
 #include<sstream>
class Codec {
public:
    // Encodes a tree to a single string.
    string serialize(TreeNode* root) {
        if(!root)
            return "";
        string str;

        queue<TreeNode*> q;
        q.push(root);

        while(!q.empty())
        {
            TreeNode* tmp = q.front();
            q.pop();

            if(tmp == nullptr)
            {
                str = str + "#,";
            }
            else
                str = str + to_string(tmp->val) + ',';

            if(tmp != nullptr)
            {
                q.push(tmp->left);
                q.push(tmp->right);
            }
        }
        cout <<str;
        return str;
    }

    // Decodes your encoded data to tree.
    TreeNode* deserialize(string data) {
        if(data.size() == 0)
            return nullptr;

        stringstream ss(data);

        string str;
        getline(ss, str, ',');

        TreeNode* root = new TreeNode(stoi(str));
        queue<TreeNode*> q;
        q.push(root);

        while(!q.empty())
        {
            TreeNode* tmp = q.front();
            q.pop();

            getline(ss, str, ',');

            if(str == "#")
            {
                tmp->left = nullptr;
            }
            else
            {
                TreeNode* left = new TreeNode(stoi(str));
                tmp->left = left;
                q.push(left);
            }

            getline(ss, str, ',');

            if(str == "#")
            {
                tmp->right = nullptr;
            }
            else
            {
                TreeNode* right = new TreeNode(stoi(str));
                tmp->right = right;
                q.push(right);
            }
        }
        return root;
    }
};
