Construct a Binary Tree from Postorder and PreOrder Traversal

C++ Solution


 /**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
class Solution {
public:
    TreeNode* build_tree(vector& preorder,int i, int j, vector& postorder, int p, int q, map &m)
    {
        if(i > j || p > q)
            return nullptr;
        
        TreeNode* root = new TreeNode(preorder[i]);
        if(i==j)
            return root;

        int leftRootNodeVal = preorder[i+1];
        int leftRootIndexInPost = m[leftRootNodeVal];

        int size = leftRootIndexInPost - p + 1;

        root->left = build_tree(preorder, i+1, i+size, postorder, p, p+size-1, m); 
        root->right = build_tree(preorder, i+size+1, j, postorder, p+size, q-1, m);

        return root;
    }
    TreeNode* constructFromPrePost(vector& preorder, vector& postorder) {
        map m;

        int n = postorder.size();
        for(int i = 0; i < n; i++)
            m[postorder[i]] = i;

        return build_tree(preorder, 0, n-1, postorder, 0, n-1, m);    
    }
};
