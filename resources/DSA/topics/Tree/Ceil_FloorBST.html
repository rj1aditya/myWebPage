Ceil & Floor value of BST

C++ Solution


// C++ solution code goes here
class Solution{

    public:
    int floor(Node* root, int x) {
        // Code here
        int floor = -1;
        if(root==nullptr)
            return floor;
            
        Node* cur = root;
        
        while(cur)
        {
            if(x == cur->data)
                return x;
            if(x > cur->data)
            {
                floor = cur->data;
                cur = cur->right;
            }
            else
                cur = cur->left;
        }
        return floor;
    }

    int findCeil(Node* root, int input) {
        int ceil = -1;
        if (root == NULL) return ceil;

        Node* tmp = root;

        while(tmp)
        {
            if(tmp->data == input)
                return input;
            if(input > tmp->data)
            {
                tmp = tmp->right;
            }
            else
            {
                ceil = tmp->data;
                tmp = tmp->left;
            }

        }
        return ceil;
        // Your code here
    }
};
