Vertical Order Traversal of Binary Tree

/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
class Solution {
public:
    vector<vector<int>> verticalTraversal(TreeNode* root) {
        queue<pair<TreeNode*,pair<int,int>>> q;
        q.push({root, {0,0}});

        //vertical Key -> level order -> values
        map<int, map<int, multiset<int>>> nodes;
        vector<vector<int>> ans;

        while(!q.empty())
        {
            auto tmp = q.front();
            TreeNode* curN = tmp.first;
            q.pop();
            int vertical = tmp.second.first;
            int level = tmp.second.second;
            nodes[vertical][level].insert(curN->val);

            if(curN->left)
            {
                q.push({curN->left,{vertical-1,level+1}});
            }
            if(curN->right)
            {
                q.push({curN->right,{vertical+1,level+1}});
            }
        }

        for(auto ver: nodes)
        {
            vector<int> tmp;
            for(auto lev: ver.second)
            {
                for(auto lev1: lev.second)
                    tmp.push_back(lev1);
            }
            ans.push_back(tmp);
        }

        return ans;
    }
};
