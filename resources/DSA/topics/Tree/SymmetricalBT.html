Check for Symmetrical Binary Trees

C++ Solution


// C++ solution code goes here
class Solution{
    bool isSymm(struct Node* n1, struct Node* n2)
    {
        if(n1==nullptr || n2==nullptr)
            return n1==n2;
        
        if(n1->data != n2->data)
            return 0;

        int x = isSymm(n1->left, n2->right);

        int y = isSymm(n1->right, n2->left);

        return x&y;
    }
    public:
    // return true/false denoting whether the tree is Symmetric or not
    bool isSymmetric(struct Node* root)
    {
	    // Code here
	    return isSymm(root,root);
    }
};
