struct Node {
    int data;
    Node *left;
    Node *right;

    Node(int val) {
        data = val;
        left = right = NULL;
    }
};
*/
class Solution {
  public:
    int minTimeToBurn(map<Node*,Node*> &m, Node* tar)
    {
        map<Node*, int> vis;
        vis[tar] = 1;
        
        queue<Node*> q;
        q.push(tar);
        int ans = 0;
        while(!q.empty())
        {
            int size = q.size();
            int burn = 0;
            for(int i = 0; i < size; i++)
            {
                Node* tmp = q.front();
                q.pop();
                if(tmp->left && vis[tmp->left] != 1)
                {
                    vis[tmp->left] = 1;
                    q.push(tmp->left);
                    burn = 1;
                }
                if(tmp->right && vis[tmp->right] != 1)
                {
                    vis[tmp->right] = 1;
                    q.push(tmp->right);
                    burn = 1;
                }
                if(m[tmp] && vis[m[tmp]] != 1)
                {
                    vis[m[tmp]] = 1;
                    q.push(m[tmp]);
                    burn = 1;
                }
            }
            if(burn)
                ans++;
        }
        return ans;
    }
    
    Node* mapParentNodes(Node* root, int target, map<Node*, Node*> &m)
    {
        queue<Node*> q;
        q.push(root);
        Node* ret = nullptr;
        while(!q.empty())
        {
            Node* tmp = q.front();
            q.pop();
            
            if(tmp->data == target)
                ret = tmp;
            
            if(tmp->left)
            {
                m[tmp->left] = tmp;
                q.push(tmp->left);
            }
            if(tmp->right)
            {
                m[tmp->right] = tmp;
                q.push(tmp->right);
            }
        }
        return ret;
    }
    int minTime(Node* root, int target) 
    {
        // Your code goes here
        map<Node*, Node*> m;
        Node* tar = mapParentNodes(root, target, m);
        
        return minTimeToBurn(m, tar);
    }
};
