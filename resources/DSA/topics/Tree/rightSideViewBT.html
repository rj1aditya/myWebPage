/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
class Solution {
    void preOrderRecursion(TreeNode* node, vector<int> &ans, int lvl)
    {
        if(node==nullptr)
            return;
        
        if(ans.size() == lvl)
            ans.push_back(node->val);

        preOrderRecursion(node->right, ans, lvl+1);        
        preOrderRecursion(node->left, ans, lvl+1);
    }

    void lvlOrderTraverse(TreeNode* node, vector<int> &ans, int lvl)
    {
        if(node==nullptr)
            return;
        
        queue<pair<TreeNode*,int>> q;
        q.push({node, lvl});

        while(!q.empty())
        {
            TreeNode* tmp = q.front().first;
            int lvl = q.front().second;
            q.pop();

            if(ans.size() == lvl)
                ans.push_back(tmp->val);

            if(tmp->right)
                q.push({tmp->right,lvl+1});
            if(tmp->left)
                q.push({tmp->left,lvl+1});
        }
    }
public:
    vector<int> rightSideView(TreeNode* root) {
        vector<int> ans;
        //preOrderRecursion(root, ans, 0);
        lvlOrderTraverse(root, ans, 0);
        return ans;   
    }
};
