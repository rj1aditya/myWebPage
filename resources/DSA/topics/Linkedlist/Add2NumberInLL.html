Node *addTwoNumbers(Node *num1, Node *num2)
{
    // Write your code here.
    int carry = 0;
    
    Node *dummy = new Node(-1);
    Node *curr = dummy;
    while(num1 || num2)
    {
        int sum = 0;
        sum += carry;
        if(num1)
        {
            sum += num1->data;
            num1 = num1->next;
        }
        if(num2)
        {
            sum += num2->data;
            num2 = num2->next;
        }
        Node *newN = new Node(sum%10);
        carry = sum/10;

        curr->next = newN;
        curr = newN;
    }
    if(carry)
    {
        Node *newN = new Node(carry);
        curr->next = newN;
        curr = newN;
    }

    return dummy->next;
}
