class Node:
    def __init__(self, data=0, next=None):
        self.val = data
        self.next = next


# Please do not change code above.

def findLength(slw, fst):
    cnt = 1
    while slw is not fst:
        cnt += 1
        slw = slw.next

    return cnt

def lengthOfLoop(head: Node) -> int:
    # Write your code here
    fast = head
    slow = head
    
    while fast and fast.next:
        fast = fast.next.next
        slow = slow.next

        if slow is fast:
            return findLength(slow.next, fast)

    return 0
