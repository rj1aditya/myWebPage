vector<pair<int, int>> findPairs(Node* head, int sum)
{
    // Write your code here.
    vector<pair<int, int>> ans;
    //find tail of the DLL
    Node* temp2 = head;
    while(temp2->next)
    {
        temp2 = temp2->next;
    }
    Node* temp1 = head;
    
    while(temp1->data < temp2->data)
    {
        int add = temp1->data+temp2->data;
        //cout <<temp1->data<<" "<<temp2->data<<" "<<add<<endl;
        if( add == sum)
        {
            ans.push_back({temp1->data,temp2->data});
            temp2 = temp2->prev;
            temp1 = temp1->next;
        }
        else if(add < sum)
            temp1 = temp1->next;
        else
            temp2 = temp2->prev;
    }

    return ans;
}
