Node * removeDuplicates(Node *head)
{
    // Write your code here
    if( !head || !head->next)
        return head;

    Node *temp = head;

    while(temp && temp->next)
    {
        if(temp->data == temp->next->data)
        {
            Node *duplicate = temp->next;
            temp->next = duplicate->next;
            if(duplicate->next)
                duplicate->next->prev = temp;
            
            delete duplicate;
        }
        else
            temp = temp->next;
    }
    return head;
}
