Design a Browser History

class Node
{
    public:
    Node* next;
    Node* prev;
    string data;
    Node() : next(nullptr), prev(nullptr), data("") {}
    Node(string url) : next(nullptr), prev(nullptr), data(url) {}
    Node(string url, Node* prv, Node* nxt) : next(nxt), prev(prv), data(url) 
    {}
};

class Browser
{
    Node *currPage;
    public:
    Browser(string &homepage)
    {
        // Write you code here
        currPage = new Node(homepage);
    }

    void visit(string &url)
    {
        // Write you code here
        Node *newPage = new Node(url);
        currPage->next = newPage;
        newPage->prev = currPage;
        currPage = newPage;
    }

    string back(int steps)
    {
        // Write you code here
        while(steps-- && currPage->prev != nullptr)
        {
            currPage = currPage->prev;
        }

        return currPage->data;
    }

    string forward(int steps)
    {
        // Write you code here
        while(steps-- && currPage->next != nullptr)
        {
            currPage = currPage->next;
        }
        return currPage->data;
    }
};
