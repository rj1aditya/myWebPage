Rotate a LinkedList

C++ Solution


// C++ solution code goes here
class Solution {
public:
    ListNode* rotateRight(ListNode* head, int k) {
        
        if(head == nullptr)
            return head;

        int len = 1;
        ListNode* lastN = head;
        while( lastN->next )
        {
            len++;
            lastN = lastN->next;
        }
        
        int len2 = k%len;

        if(len2 == 0)
            return head;
        
        int iterate = len - len2;
        
        ListNode* temp = head;
        while(iterate>1)
        {
            iterate--;
            temp = temp->next;
        }
        
        lastN->next = head;
        head = temp->next;
        temp->next = NULL;

        return head;
            
    }
};
