        self.data = data
        self.next = None

class list:

        self.head = None

    def append(self, val):
        newN = Node(val)
        if self.head is None:
            self.head = newN
        else:
            curN = self.head
            while curN.next is not None:
                curN = curN.next
            curN.next = newN

    def display(self):
        curN = self.head
        while curN is not None:
            print(curN.data, end=' ')
            curN = curN.next
            
def reverseLinkedList(head):
    # write your code here
    if head is None or head.next is None:
        return head
    
    newHead = reverseLinkedList(head.next)
    frontNode = head.next
    frontNode.next = head
    head.next = None

    return newHead

    # write your code here
    fastN = head
    slowN = head

    while fastN.next is not None and fastN.next.next is not None:
        fastN = fastN.next.next
        slowN = slowN.next

    firstN = head
    newHead = reverseLinkedList(slowN.next)
    secondN = newHead
    print(id(newHead))
    ans = True
    while secondN is not None:
        if secondN.data != firstN.data:
            ans = False
            break
        secondN = secondN.next
        firstN = firstN.next

    print(id(fastN.next))

    return ans
