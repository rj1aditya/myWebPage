/****************************************************************

    Following is the class structure of the Node class:

        class Node
        {
        public:
	        int data;
	        Node *next;
	        Node(int data)
	        {
		        this->data = data;
		        this->next = NULL;
	        }
        };

Node* findMiddleNode(Node *head)
{
	if(!head || !head->next)
		return head;

	Node *fastN = head->next;
	Node *slowN = head;

	while(fastN && fastN->next)
	{
		slowN = slowN->next;
		fastN = fastN->next->next;
	}

	return slowN;
}

Node* mergetIt(Node *l1, Node *l2)
{
	Node *dummy = new Node(-1);
	Node *temp = dummy;
	while( l1 && l2 )
	{
		if(l1->data < l2->data)
		{
			temp->next = l1;
			l1 = l1->next;
		}
		else
		{
			temp->next = l2;
			l2 = l2->next;
		}
		temp = temp->next;
	}

	if(l1)
		temp->next = l1;
	if(l2)
		temp->next = l2;

	return dummy->next;
}

Node *sortLL(Node *head)
{
    // Write your code here.
	if(!head || !head->next)
		return head;

	Node *mid = findMiddleNode(head);
	
	Node* firstN = head;
	Node* secondN = mid->next;
	mid->next = nullptr;
	Node* leftNode = sortLL(firstN);
	Node* rightNode = sortLL(secondN);

	return mergetIt(leftNode, rightNode);
}
