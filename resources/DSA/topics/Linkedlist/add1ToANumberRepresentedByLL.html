Add 1 to a number represented by LinkedList

C++ Solution


// C++ solution code goes here
/**
 * Definition of linked list:
 *
 * class Node {
 * public:
 *      int data;
 *      Node *next;
 *      Node() {
 *          this->data = 0;
 *          this->next = NULL;
 *      }
 *      Node(int data) {
 *          this->data = data;
 *          this->next = NULL;
 *      }
 *      Node (int data, Node *next) {
 *          this->data = data;
 *          this->next = next;
 *      }
 * };
 *
 *************************************************************************/
 int helper(Node *head)
 {
     if(head == nullptr)
         return 1;
     
     int carry = helper(head->next);
 
     int val = head->data + carry;
     head->data = val%10;
     carry = val/10;
     return carry;
 }
 Node *addOne(Node *head)
 {
     // Write Your Code Here.
     if( helper(head) == 1)
     {
         Node *newN = new Node(1);
         newN->next = head;
         head = newN;
     }
 
     return head;
 }
