class Node
{
public:
    int data;
    Node *next;
    Node()
    {
        this->data = 0;
        next = NULL;
    }
    Node(int data)
    {
        this->data = data; 
        this->next = NULL;
    }
    Node(int data, Node* next)
    {
        this->data = data;
        this->next = next;
    }
};

Node* sortList(Node *head){
    // Write your code here.

    Node *zeroH = new Node(-1); Node *zeroN = zeroH;
    Node *oneH = new Node(-1); Node *oneN = oneH;
    Node *twoH = new Node(-1); Node *twoN = twoH;

    Node *temp = head;
    while(temp != nullptr)
    {
        if(temp->data == 0)
        {
            zeroN->next = temp;
            zeroN = temp;
        }
        else if(temp->data == 1)
        {
            oneN->next = temp;
            oneN = temp;
        }
        else{
            twoN->next = temp;
            twoN = temp;
        }
        temp = temp->next;
    }

    zeroN->next = oneH->next ? oneH->next : twoH->next;
    oneN->next = twoH->next;
    twoN->next = nullptr;//This line is important, else it cause LTE
    return zeroH->next;
}
