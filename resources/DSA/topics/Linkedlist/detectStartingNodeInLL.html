C++ Solution


// C++ solution code goes here
//Approach first using map

#include<bits/stdc++.h>
Node *firstNode(Node *head)
{
    //    Write your code here.

    std::map<Node*, int> m;

    Node *cur = head;
    while(cur)
    {
        if(m.find(cur) != m.end())
            return cur;
        m[cur] = 1;
        cur = cur->next;
    }

    return nullptr;
}

//optimized Approach
#include<bits/stdc++.h>
Node *firstNode(Node *head)
{
    //Write your code here.
    Node* fast = head;
    Node* slow = head;

    while(fast && fast->next)
    {
        fast = fast->next->next;
        slow = slow->next;

        //2nd step
        if(fast == slow)
        {
            slow = head;
            while(slow != fast)
            {
                slow = slow->next;
                fast = fast->next;
            }
            return slow;
        }
    }
    return nullptr;
}
