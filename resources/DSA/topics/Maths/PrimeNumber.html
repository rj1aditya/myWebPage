class Solution {
    public:
      bool isPrime(int n) {
          // code here
          int cnt = 0;
          for(int i = 1; i*i <= n; i++)
          {
              int factor = i;
              int otherFactor = n/i;
              
              if(n%factor == 0)
                  cnt++;
              if(factor != otherFactor && n%otherFactor == 0)
                  cnt++;
              
              if(cnt > 2)
                  break;
          }
          
          if(cnt == 2)
              return true;
          
          return false;
      }
  };
