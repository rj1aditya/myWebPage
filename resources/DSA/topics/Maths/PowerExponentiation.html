C++ Solution


// C++ solution code goes here
class Solution {
    public:
        double myPow(double x, int n) {
            double ans = 1.0;
            long m = abs(long(n));
            while(m >= 1)
            {
                //if n is odd number
                if( m & 1 )
                {
                    ans = ans* x;
                    m = m-1;
                }
                else //if n is even number
                {
                    x = x*x;
                    m = m/2;
                }
            }
            if(n < 0)
                return 1/ans;
        
            return ans;
    
        }
    };
