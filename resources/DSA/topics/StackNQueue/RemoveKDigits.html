Remove K Digits

C++ Solution


// C++ solution code goes here
class Solution {
    public:
        string removeKdigits(string num, int k) {
            int n = num.length();
            //Edge cases 1
            if(k == n)
                return "0";
    
            //Edge case 2, where string could be 0001325
    
    
    
            stack<char> st;
            for(int i = 0; i < n; i++)
            {
                while(!st.empty() && k > 0 && st.top()-'0' > num[i]-'0')
                {
                    st.pop();
                    k--;
                }
                st.push(num[i]);
            }
    
            //Edgecase if string is 123456, k = 3
            while(k>0)
            {
                st.pop();
                k--;
            }
    
            string ans;
            while(!st.empty())
            {
                ans += st.top();
                st.pop();
            }
    
            //Edge case if string is 00012382
            while(ans.length() > 1 && ans.back() == '0')
                ans.pop_back();
    
            reverse(ans.begin(), ans.end());
            return ans;
    
        }
    };
