Largest Rectangle in Histogram

C++ Solution


// C++ solution code goes here
class Solution {
  public:
      int largestRectangleArea(vector<int>& heights) {
          stack<int> st;
          int maxi = 0;
          int n = heights.size();
  
          // Push a sentinel index to mark the base
          st.push(-1);
  
          for (int i = 0; i < n; i++) {
              // Process all bars that are taller than the current bar
              while (st.top() != -1 && heights[i] < heights[st.top()]) {
                  int ind = st.top();
                  st.pop();
                  int height = heights[ind];
                  int width = i - st.top() - 1; // Calculate width
                  maxi = max(maxi, height * width);
              }
              st.push(i);
          }
  
          // Process the remaining bars in the stack
          while (st.top() != -1) {
              int ind = st.top();
              st.pop();
              int height = heights[ind];
              int width = n - st.top() - 1; // Calculate width
              maxi = max(maxi, height * width);
          }
  
          return maxi;
      }
  };
