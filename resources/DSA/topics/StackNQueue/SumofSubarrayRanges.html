Sum of Subarray Ranges

class Solution {
    vector<int> nse(vector<int>& nums)
    {
        stack<int> st;
        int n = nums.size();
        vector<int> ans(n, n);
        for(int i  = n-1; i >= 0; i--)
        {
            while(!st.empty() && nums[st.top()] >= nums[i])
                st.pop();
            
            if(!st.empty())
                ans[i] = st.top();

            st.push(i); 
        }
        return ans;
    }
    vector<int> psee(vector<int>& nums)
    {
        stack<int> st;
        int n = nums.size();
        vector<int> ans(n, -1);
        for(int i  = 0; i < n ; i++)
        {
            while(!st.empty() && nums[st.top()] > nums[i])
                st.pop();
            
            if(!st.empty())
                ans[i] = st.top();

            st.push(i); 
        }
        return ans;
    }

    vector<int> nme(vector<int>& nums)
    {
        stack<int> st;
        int n = nums.size();
        vector<int> ans(n, n);
        for(int i  = n-1; i >= 0; i--)
        {
            while(!st.empty() && nums[st.top()] <= nums[i])
                st.pop();
            
            if(!st.empty())
                ans[i] = st.top();

            st.push(i); 
        }
        return ans;
    }
    vector<int> pmee(vector<int>& nums)
    {
        stack<int> st;
        int n = nums.size();
        vector<int> ans(n, -1);
        for(int i  = 0; i < n ; i++)
        {
            while(!st.empty() && nums[st.top()] < nums[i])
                st.pop();
            
            if(!st.empty())
                ans[i] = st.top();

            st.push(i); 
        }
        return ans;
    }
public:
    long long subArrayRanges(vector<int>& nums) {

        int n = nums.size();
        vector<int> p = nse(nums);
        vector<int> q = psee(nums);
        vector<int> r = nme(nums);
        vector<int> t = pmee(nums);

        long long largestSum = 0;
        long long smallestSum = 0;

        //finding smallestsum
        for(int i = 0; i < n; i++)
        {
            long long a = p[i] - i;
            long long b = i - q[i];
            smallestSum = (smallestSum + (a*b)*nums[i]);
        }

        //finding largestsum
        for(int i = 0; i < n; i++)
        {
            long long a = r[i] - i;
            long long b = i - t[i];
            largestSum = (largestSum + (a*b)*nums[i]);
        }

        return largestSum-smallestSum;
    }
};
