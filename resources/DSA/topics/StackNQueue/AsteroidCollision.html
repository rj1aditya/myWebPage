Asteroid Collision

C++ Solution


// C++ solution code goes here
class Solution {
public:
vector<int> asteroidCollision(vector<int>& asteroids) {
    //we should use stack, but we are using vector here as a stack 
    //so that at the end we not need to reverse the left elems
    //in the stack and return as ans.
    vector<int> st;
    int n = asteroids.size();
    for(int i = 0; i < n; i++)
    {
        if(asteroids[i] > 0)
        {
            st.push_back(asteroids[i]);
        }
        else
        {
           while(!st.empty() && st.back() > 0 && st.back() < abs(asteroids[i]))
                st.pop_back();        
            if(!st.empty() && st.back() ==  abs(asteroids[i]) )
                st.pop_back();
            else if(st.empty() || st.back() < 0)
                st.push_back( asteroids[i]); 
        }
    }
    return st;
}
};
