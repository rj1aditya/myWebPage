class Solution {

    int findMaximumRectangleInHistogram(vector<int> matrix)
    {
        stack<int> st;
        st.push(-1);
        int maxi = 0;
        for(int i = 0; i < matrix.size(); i++)
        {
            while(st.top() != -1 && matrix[st.top()] > matrix[i])
            {
                int ind = st.top();
                st.pop();
                int width = i - st.top() - 1;
                maxi = max(maxi, matrix[ind]*width);
            }
            st.push(i);
        }
    
        while(st.top() != -1)
        {
            int ind = st.top();
            st.pop();
            int width = matrix.size() - st.top() - 1;
            maxi = max(maxi, matrix[ind]*width);
        }
    
        return maxi;
    }
    
    public:
        int maximalRectangle(vector<vector<char>>& matrix) {
            int n = matrix.size();
            int m = matrix[0].size();
            int maxi = 0;
            //making prefixsum
            vector<vector<int>> newM(n, vector<int>(m, 0));
            for(int j = 0; j < m; j++)
            {
                int sum = 0;
                for(int i = 0; i < n; i++)
                {
                    sum += matrix[i][j] - '0';
                    if(matrix[i][j] == '0')
                        sum = 0;
                    
                    newM[i][j] = sum;
                }
            }
    
            for(int i = 0; i < n; i++)
            {
                maxi = max(maxi, findMaximumRectangleInHistogram(newM[i]));
            }
    
            return maxi;
        }
    };
