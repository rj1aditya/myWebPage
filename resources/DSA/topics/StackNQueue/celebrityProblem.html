The Celebrity Problem

C++ Solution


// C++ solution code goes here
class Solution {
    public:
      // Function to find if there is a celebrity in the party or not.
      int celebrity(vector<vector<int> >& mat) {
          // code here
          int top = 0;
          int down = mat.size()-1;
          
          
          while(top < down)
          {
              if(mat[top][down] == 1 && mat[down][top] == 1)
              {
                  top++;
                  down--;
              }
              else if(mat[top][down] == 1)
                  top++;
              else
                  down--;
          }
          
          if(top > down)
              return -1;
          else
          {
              //horizontal should be all 0
              for(int i = 0; i < mat[0].size(); i++)
              {
                  if(mat[top][i] != 0)
                      return -1;
              }
              //verticle should be all 1 execpt [n][n] cell
              for(int i = 0; i < mat.size(); i++)
              {
                  if(i==top)
                      continue;
  
                  if(mat[i][top] != 1)
                      return -1;
              }
              
              return top;
          }
          return -1;   
      }
  };
