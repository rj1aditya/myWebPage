Reverse array and check string is palindrome

C++ Solution

//Reverse an array
class solution
{
    public:
    void reverse(int i, int *arr, int n)
    {
        if(i >= n)
            return;
        
        swap(arr[i], arr[n-1]);
        reverse(i+1, arr, n-1);
    }
};

class checkPalindrome {

    public static Boolean isPalindrome(String s, int i)
    {
        if(i > s.length()/2)
            return true;
        
        if(s.charAt(i) != s.charAt(s.length()-i-1))
            return false;

        return isPalindrome(s, i+1);
    }
    public static void main(String args[])
    {
        if(args.length == 0)
        {
            System.out.println("Insuffient input, pass input string from cmd line\n");
            return;
        }
        String s="";
        for(int i = 0; i < args.length; i++)
            s = s+args[i];
        
        if(isPalindrome(s, 0))
            System.out.println("Palindrome");
        else
            System.out.println("Not Palindrome");
    }
}
