Print all permutation of an array

C++ Solution


// C++ solution code goes here
//without map
void fun(vector<int> &arr, int indx, vector<vector<int>> &ans, int n)
{
    if (indx == n)
    {
        ans.push_back(arr);
        return;
    }

    for (int i = indx; i < n; i++)
    {
        swap(arr[i], arr[indx]);
        fun(arr, indx + 1, ans, n);
        swap(arr[i], arr[indx]);
    }
}

Java Solution


// Java solution code goes here
//Using with Map
import java.util.*;
class solution
{
    public void fun(int []arr, List<Integer> ds, List<List<Integer>> res, boolean []map)
    {
        if(ds.size() == arr.length)
        {
            res.add(new ArrayList<>(ds));
            return;
        }

        for(int i = 0; i < arr.length; i++)
        {
            if(map[i] == false)
            {
                map[i] = true;
                ds.add(arr[i]);
                fun(arr, ds, res, map);
                map[i] = false;
                ds.remove(ds.size()-1);
            }
        }
    }
    public List<List<Integer>> permutation(int []arr)
    {
        boolean []map = new boolean[arr.length];
        List<List<Integer>> res = new ArrayList<>();
        fun(arr, new ArrayList<>(), res, map);
        return res;
    }
}
