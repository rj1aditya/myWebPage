//Print All subsequences of an array whose sum is K
class solution
{
    public:
    void printSub(int arr[], vector &ds, int sum, int k, int idx, int n)
    {
        if(sum > k)
            return;
        if(idx == n)
        {
            if(sum == k)
            {
                for(auto s: ds)
                    cout<< s << " ";
                cout << endl;
            }
            return;
        }
        ds.push_back(arr[idx]);
        printSub(arr, ds, sum+arr[idx], k, idx+1, n);
        ds.pop_back();
        printSub(arr, ds, sum, k, idx+1, n);
    }
};

//Print ANY subsequences of an array whose sum is K
class solution2
{
    public:
    bool printSub(int arr[], vector &ds, int sum, int k, int idx, int n)
    {
        if(sum > k)
            return false;
        if(idx == n)
        {
            if(sum == k)
            {
                for(auto s: ds)
                    cout<< s << " ";
                return true;
            }
            return false;
        }
        ds.push_back(arr[idx]);
        if(printSub(arr, ds, sum+arr[idx], k, idx+1, n) == true)
            return true;
        ds.pop_back();
        if(printSub(arr, ds, sum, k, idx+1, n) == true)
            return true;
        
        return false;
    }
};

//Print COUNT subsequences of an array whose sum is K
class solution
{
    public:
    int printSub(int arr[], vector &ds, int sum, int k, int idx, int n)
    {
        if(sum > k)
            return 0;
        if(idx == n)
        {
            if(sum == k)
            {
                return 1;
            }
            return 0;
        }
        ds.push_back(arr[idx]);
        int l = printSub(arr, ds, sum+arr[idx], k, idx+1, n);
        ds.pop_back();
        int r = printSub(arr, ds, sum, k, idx+1, n);
        return l+r;
    }
};
