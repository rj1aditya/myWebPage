class Solution 
{
    bool isSafe(int row, int col, vector<string> &dp, int n)
    {
        int dupR = row;
        int dupC = col;
        
        while(row >= 0 && col >= 0)
        {
            if(dp[row][col] == 'Q')
                return false;
            
            row--;
            col--;
        }
        
        
        row = dupR;
        col = dupC;
        while(col >= 0)
        {
            if(dp[row][col] == 'Q')
                return false;

            col--;
        }
        
        row = dupR;
        col = dupC;
        while(row < n && col >= 0)
        {
            if(dp[row][col] == 'Q')
                return false;
            
            row++;
            col--;
        }
        
        return true;
    }
    
    void findAns(vector<vector<string>> &ans, vector<string> &dp, int col, int N , vector<int> &left, vector<int> &lowerDiag, vector<int> &upperDiag)
    {
        if(col == N)
        {
            ans.push_back(dp);
            return;
        }
        
        for(int r = 0; r < N; r++)
        {
        if (left[r] == 1 || lowerDiag[r + col] == 1 || upperDiag[N - 1 + col - r] == 1)
            continue;

        //Setting
        left[r] =1; 
        lowerDiag[r + col] = 1; 
        upperDiag[N - 1 + col - r] = 1;
        dp[r][col] = 'Q';
        findAns(ans, dp, col + 1, N, left, lowerDiag, upperDiag);

        //Resetting
        dp[r][col] = '.';
        left[r] = 0; 
        lowerDiag[r + col] = 0; 
        upperDiag[N - 1 + col - r] = 0;
        }
    }
public:
    vector<vector<string>> solveNQueens(int N) {
        string s(N, '.');
        vector<vector<string>> ans;
        vector<string> dp(N);

        vector<int> left(N, 0), lowerDiag(2 * N - 1, 0), upperDiag(2 * N - 1, 0);
        
        for(int i = 0; i < N; i++)
            dp[i] = s;
        
        findAns(ans, dp, 0, N, left, lowerDiag, upperDiag);
        return ans;
    }
};
