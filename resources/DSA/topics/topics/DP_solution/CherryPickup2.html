#include <bits/stdc++.h> 
int func(vector<vector<int>> &grid, vector<vector<vector<int>>> &dp, int i, int j1, int j2)
{
    if(j1 < 0 || j1 >= grid[0].size() || j2 < 0 || j2 >= grid[0].size())
        return -1e9;
    
    if(i == grid.size()-1)
    {
        if(j1 == j2)
            return grid[i][j1];
        else
            return grid[i][j1] + grid[i][j2];
    }
    if(dp[i][j1][j2] != -1)
        return dp[i][j1][j2];

    int maxi = -1e9;
    for(int dgj1 = -1; dgj1 <= 1; dgj1++)
    {
        for(int dgj2 = -1; dgj2 <= 1; dgj2++)
        {
            int value=0;
            if(j1 == j2)
                value = grid[i][j1];
            else
                value = grid[i][j1] + grid[i][j2];
            value += func(grid, dp, i+1, j1+dgj1, j2+dgj2);

            maxi = max(maxi, value);
        }
    }
    return dp[i][j1][j2] = maxi;
}

    vector<vector<vector<int>>> dp(n, vector<vector<int>>(m, vector<int>(m, -1)));

    return func(grid, dp,0,  0, m-1);
}
