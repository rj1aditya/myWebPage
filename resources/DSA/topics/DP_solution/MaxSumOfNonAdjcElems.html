Maximum Sum of Non-Adjacent Elements

C++ Solution


// C++ solution code goes here
#include <bits/stdc++.h> 

//recursion
int func(vector<int> &nums, vector<int> &dp, int indx)
{
    if(indx == 0)
        return nums[indx];
    if(indx < 0)
        return 0;

    if(dp[indx] != -1)
        return dp[indx];

    int pick = nums[indx] + func(nums, dp, indx-2);
    int unpick = 0 + func(nums, dp, indx-1);

    return dp[indx] = max(pick, unpick);
}

//tabulation
int func2(vector<int> &nums, vector<int> &dp)
{
    dp[0] = nums[0];

    for(int i = 1; i < nums.size(); i++)
    {
        int pick = nums[i];
        if(i-2 >= 0)
            pick += dp[i-2];
        
        int unpick = dp[i-1];

        dp[i] = max(pick, unpick);
    }
    return dp[nums.size()-1];
}

//space optimization
int func3(vector<int> &nums)
{
    int prev = nums[0];
    int prev2 = 0;

    for(int i = 1; i < nums.size(); i++)
    {
        int pick = nums[i];
        if(i-2 >= 0)
            pick += prev2;
        
        int unpick = prev;

        int cur = max(pick, unpick);
        prev2 = prev;
        prev = cur;
    }
    return prev;
}
