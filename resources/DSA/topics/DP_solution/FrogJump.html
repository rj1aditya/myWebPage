#include <bits/stdc++.h> 
int fun(vector<int> &h, vector<int> &dp, int indx)
{
    if(indx == 0)
        return 0;
    
    if(dp[indx] != -1)
        return dp[indx];
    int left = fun(h, dp, indx-1) + abs(h[indx] - h[indx-1]);
    int right = INT_MAX;
    if(indx > 1)
        right = fun(h, dp, indx-2) + abs(h[indx] - h[indx-2]);
    return dp[indx] = min(left, right);
}

int frogJump(int n, vector<int> &heights)
{

    vector<int> dp(n+1,-1);
    return fun(heights, dp, n-1);
}

//Tabulation solution
int frogJump(int n, vector<int> &heights)
{
    // Write your code here.
    vector<int> dp(n,-1);
    dp[0] = 0;

    for(int i = 1; i < n; i++)
    {
        int left = dp[i-1] + abs(heights[i] - heights[i-1]);
        int right = 1e9+7;
        if(i>1)
            right = dp[i-2] + abs(heights[i] - heights[i-2]);

        dp[i] = min(left) + right;
    }
    return dp[n-1];
}

    for(int i = 1; i < n; i++)
    {
        int left = prev + abs(heights[i] - heights[i-1]);
        int right = 1e9+7;
        if(i>1)
            right = prev2 + abs(heights[i] - heights[i-2]);

        int cur = min(left,right);
        prev2 = prev;
        prev = cur;
    }
    return prev;
}
