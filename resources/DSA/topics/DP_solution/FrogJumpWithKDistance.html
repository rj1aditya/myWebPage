Frog jump with K distance

C++ Solution


// C++ solution code goes here
#include <iostream>
#include <vector>
#include <cstdlib>
using namespace std;

class Solution
{
    // Recursive approach
    int fun(vector<int> &h, vector<int> &dp, int k, int n, int indx)
    {
        if (indx == 0)
            return 0;

        if (dp[indx] != -1)
            return dp[indx];

        int minEnergy = INT_MAX;
        for (int i = 1; i <= k; i++)
        {
            int jump = INT_MAX;
            if (indx - i >= 0)
                jump = fun(h, dp, k, n, indx - i) + abs(h[indx] - h[indx - i]);

            minEnergy = min(minEnergy, jump);
        }
        return dp[indx] = minEnergy;
    }

    //Tabulation approach
    int fun(int k, int n, vector<int> &h, vector<int> &dp)
    {
        dp[0] = 0;

        for (int i = 1; i < n; i++)
        {
            int minEnergy = INT_MAX;
            for (int j = 1; j <= k; j++)
            {
                if (i - j >= 0)
                {
                    int jump = dp[i - j] + abs(h[i] - h[i - j]);
                    minEnergy = min(minEnergy, jump);
                }
                dp[i] = minEnergy;
            }
        }

        return dp[n - 1];
    }

public:
    int findMinEnerge(vector<int> &h, int k, int n)
    {
        vector<int> dp(n, -1);
        return fun(h, dp, k, n, n - 1);
    }
    int findMinEnerge(int k, int n, vector<int> &h)
    {
        vector<int> dp(n, -1);
        return fun(k, n, h, dp);
    }
};

int main()
{
    vector<int> h{10, 30, 40, 50, 20};
    Solution obj;
    int k = 3;
    cout << obj.findMinEnerge(h, k, h.size());
    cout << endl
         << obj.findMinEnerge(k, h.size(), h);
    return 0;
}
