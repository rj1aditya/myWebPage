#include <bits/stdc++.h> 
//Recursion
int fun(vector<vector<int>> &dp, int row, int col)
{
    if(row == 0 && col == 0)
        return 1;

    if(row < 0 || col < 0)
        return 0;
    if(dp[row][col] != -1)
        return dp[row][col];
    int up = fun(dp, row-1, col);
    int left = fun(dp, row, col-1);

    return dp[row][col]=up+left;
}
//Tabulation
int fun2(vector<vector<int>> &dp, int row, int col)
{
    //Note: here we started from row,col 0,0 instead of 1,1. Coz if we started from 1,1 then entire row and col elements would be missed.
    for(int r=0; r < row; r++)
    {
        for(int c=0; c< col; c++)
        {
            int right=0;
            int down=0;
            if(r==0 && c==0)
            {
                dp[0][0] = 1;
            }
            else
            {
                if(r>0)
                    down = dp[r-1][c];
                if(c>0)
                    right = dp[r][c-1];
                dp[r][c] = right+down;
            }
        }
    }
    return dp[row-1][col-1];
}
//space optimization
int fun3(int row, int col)
{
    vector<int> temp(col, 0);
    vector<int> prevRow(col, 0);
    
    //Note: here we started from row,col 0,0 instead of 1,1. Coz if we started from 1,1 then entire row and col elements would be missed.
    for(int r=0; r < row; r++)
    {
        for(int c=0; c< col; c++)
        {
            int right=0;
            int down=0;
            if(r==0 && c==0)
            {
                right = 1;
            }
            else
            {
                if(r>0)
                    down = prevRow[c];
                if(c>0)
                    right = temp[c-1];
            }
            temp[c] = right+down;
            //cout <<temp[c]<<" ";
        }
        //cout <<endl;
        prevRow = temp;
    }
    return prevRow[col-1];
}
