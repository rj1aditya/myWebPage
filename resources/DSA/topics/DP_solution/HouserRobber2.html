#include <bits/stdc++.h> 

long long int func(vector<int> &house)
{
    long long int prev = house[0];
    long long int prev2 = 0;

    for(int i = 1; i < house.size(); i++)
    {
        long long int loot = house[i];
        if(i-2>=0)
            loot += prev2;
        
        long long int noLoot = prev;

        long long int cur = max(loot, noLoot);

        prev2 = prev;
        prev = cur;
    }

    return prev;
}


long long int houseRobber(vector<int>& valueInHouse)
{
    // Write your code here.
    int n = valueInHouse.size();
    if(n ==1)
        return valueInHouse[0];

    vector<int> temp1;
    vector<int> temp2;
    for(int i = 0; i < n; i++)
    {
        if(i != 0)
            temp1.push_back(valueInHouse[i]);
        if(i != n-1)
            temp2.push_back(valueInHouse[i]);
    }

    return max(func(temp1), func(temp2));
}
