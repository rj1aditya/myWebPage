bool check(string &s1, string &s2)
{
    int p1 = 0;
    int p2 = 0;

    while(p1 < s1.size())
    {
        if(p2 < s2.size() && s1[p1] == s2[p2])
        {
            p1++; p2++;
        }
        else
        {
            p1++;
        }
    }

    return (p1 == s1.size()) && (p2==s2.size());
}


bool comp(string &s1, string &s2)
{
    return s1.size() < s2.size();
}

int longestStrChain(vector &arr){
    // Write your code here.
    //Sorting the vector of string based on their size
    sort(arr.begin(), arr.end(), comp);

    vector dp(arr.size(), 1);

    int maxi = 1;

    for(int i = 0; i < arr.size(); i++)
    {
        for(int j = 0; j < i; j++)
        {
            if(arr[i].size() == arr[j].size()+1 && check(arr[i], arr[j]) && dp[j]+1 > dp[i] )
            {
                dp[i] = 1+ dp[j];
            }
        }
        if(maxi < dp[i])
        {
            maxi = dp[i];
        }
    }

    return maxi;
}
