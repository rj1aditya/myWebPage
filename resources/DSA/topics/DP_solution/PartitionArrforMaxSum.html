//Recursion
int fun1(vector<int> &arr, int n, vector<int> &dp, int i, int k)
{
    if(i == n)
        return 0;

    if(dp[i] != -1)
        return dp[i];

    int len = 0;
    int maxSum = INT_MIN;
    int maxElem = INT_MIN;
    for(int ind = i; ind < min(i + k, n); ind++)
    {
        maxElem = max(maxElem, arr[ind]);
        len++;
        int x = len*maxElem + fun1(arr, n, dp, ind+1, k);
        maxSum = max(maxSum, x);
    }

    return dp[i] = maxSum;
}

int fun2(vector<int> &arr, int k)
{
    int n = arr.size();
    vector<int> dp(n+1, 0); 

    for(int i= n-1; i >= 0; i--)
    {
        int len = 0;
        int maxSum = INT_MIN;
        int maxElem = INT_MIN;
        for(int ind = i; ind < min(i + k, n); ind++)
        {
            maxElem = max(maxElem, arr[ind]);
            len++;
            int x = len*maxElem + dp[ind+1];
            maxSum = max(maxSum, x);
        }
        dp[i] = maxSum;
    }
    return dp[0];
}
