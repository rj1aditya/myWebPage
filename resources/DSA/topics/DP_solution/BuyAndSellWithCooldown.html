//Recursion
int func1(vector<int> &values, int n, vector<vector<int>> &dp, int i, int buy)
{
    if(i >= n)
        return 0;

    if(dp[i][buy] != -1)
        return dp[i][buy];

    if(buy)
    {
        return dp[i][buy] = max( (-values[i] + func1(values, n, dp, i+1, 0)), 
                                                func1(values, n, dp, i+1, 1) );
    }
    else
    {
        return dp[i][buy] = max( (values[i] + func1(values, n, dp, i+2, 1)), 
                                                func1(values, n, dp, i+1, 0) );
    }
}


//Recursion --> Tabulation
int func2(vector<int> &values, int n)
{
    vector<vector<int>> dp(n+2, vector<int>(2, 0));

    dp[n][0] = 0;
    dp[n][1] = 0;

    for(int i = n-1; i >= 0; i--)
    {
        for(int buy=0; buy <= 1; buy++)
        {
            if(buy)
            {
                dp[i][buy] = max( -values[i] + dp[i+1][0], dp[i+1][1] );
            }
            else
            {
                dp[i][buy] = max( values[i] + dp[i+2][1], dp[i+1][0]);
            }
        }
    }
    return dp[0][1];

}
