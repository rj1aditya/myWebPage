int mod=1e9+7;

//Recursion
int findCount(int i, int target, vector<int> &arr, vector<vector<int>> &dp)
{
    if(i == 0)
    {
        if(target == 0 && arr[0] == 0)
            return 2;
        if(target == arr[0] || target == 0)
            return 1;
        return 0;
    }

    if(dp[i][target] != -1)
        return dp[i][target];

    //not pick case
    int count = findCount(i-1, target, arr, dp);

    //pick
    if(arr[i] <= target)
        count += findCount(i-1, target-arr[i], arr, dp);

    dp[i][target] = count%mod;
}


//Recursion -- Tabulation
int findCount1(int n, int target, vector<int> &arr)
{
    vector<vector<int>> dp(n, vector<int>(target+1, 0));

    dp[0][0] = arr[0] ?1:2;
    if(arr[0] != 0) dp[0][arr[0]] = 1;

    for(int i = 1; i < n; i++)
    {
        for(int k = 0; k <= target; k++)
        {
            //not pick case
            int count = dp[i-1][k];

            if(arr[i] <= k)
                count += dp[i-1][k-arr[i]];

            dp[i][k] = count%mod;
        }
    }
    return dp[n-1][target];

}
