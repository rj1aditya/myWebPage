//Recursion
int fun(vector<vector<int>>& tri, vector<vector<int>>& dp, int i, int j, int n)
{
	if(i == n-1)
		return tri[i][j];

	if(dp[i][j] != -1)
		return dp[i][j];

	int down = tri[i][j] + fun(tri, dp, i+1, j, n);
	int dig = tri[i][j] + fun(tri, dp, i+1, j+1, n);

	return dp[i][j] = min(down, dig);
}

//Tabulation
int fun1(vector<vector<int>>& tri, vector<vector<int>>& dp, int n)
{

	for(int j = 0; j < n; j++)
	{
		dp[n-1][j] = tri[n-1][j];
	}

	for(int i = n-2; i >= 0; i--)
	{
		for(int j = i; j >= 0; j--)
		{
			int down = tri[i][j] + dp[i+1][j];
			int dig = tri[i][j] + dp[i+1][j+1];
			dp[i][j] = min(down, dig);
		}
	}
	return dp[0][0];
}

int fun2(vector<vector<int>>& tri, int n)
{
	vector<int> cur(n, 0);
	vector<int> prev(n, 0);
	for(int j = 0; j < n; j++)
	{
		prev[j] = tri[n-1][j];
	}

	for(int i = n-2; i >= 0; i--)
	{
		for(int j = i; j >= 0; j--)
		{
			int down = tri[i][j] + prev[j];
			int dig = tri[i][j] + prev[j+1];
			cur[j] = min(down, dig);
		}
		prev = cur;
	}
	return prev[0];
}
