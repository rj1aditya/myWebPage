#include<bits/stdc++.h>

bool isPalindrome(int i, int j, string &s)
{
    while(i < j)
    {
        if(s[i] != s[j])
            return false;
        
        i++; j--;
    }
    return true;
}

//Recursive
int func1(string &str, int n,vector<int> &dp, int i)
{
    if(i == n) return 0;
    if(dp[i] != -1)
        return dp[i];
        
    int mini = INT_MAX;
    for(int j = i; j < n; j++)
    {
        if( isPalindrome(i, j, str) )
        {
            int cost = 1 + func1(str, n, dp, j+1);
            mini = min(mini, cost);
        }
    }

    return dp[i] = mini;
}

//Recursive -- Tabulation
int func2(string &str)
{
    int n = str.size();
    vector<int> dp(n+1, 0);

    for(int i = n-1; i >= 0; i--)
    {
        int mini = INT_MAX;

        for(int j = i; j < n; j++)
        {
            if( isPalindrome(i, j, str) )
            {
                int cost = 1 + dp[j+1];
                mini = min(mini, cost);
            }
        }
        dp[i] = mini;
    }    
    return dp[0];
}
