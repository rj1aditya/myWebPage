Minimum insertions to make a string palindrome

C++ Solution

int func4(string s, string t)
{
	int n= s.length();
	int m = t.length();

	vector<int> prev(m+1,0), cur(m+1, 0);

	for(int j = 0; j <= m; j++)
		prev[j] = 0;

	for(int i = 1; i <= n; i++)
	{
		for(int j = 1; j <= m; j++)
		{
			if(s[i-1] == t[j-1])
				cur[j] = 1 + prev[j-1];
			else
				cur[j] = max(prev[j], cur[j-1]);
		}
		prev = cur;
	}

	return prev[m];
}

int minimumInsertions(string &str1)
{
	//Write your code here
    string  str2 = str1;
    reverse(str1.begin(), str1.end());

	//tabulation--SO
	return str1.length() - func4(str1, str2);
}
