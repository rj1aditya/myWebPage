Minimum Cost to Cut the Stick

C++ Solution

#include <bits/stdc++.h> 

//Recursion
int func1(vector<vector<int>> &dp, int i, int j, vector<int> &cuts)
{
    if(i > j)
        return 0;

    if(dp[i][j] != -1)
        return dp[i][j];

    int mini = INT_MAX;
    for(int ind = i; ind <= j; ind++)
    {
        int x = cuts[j+1] - cuts[i-1] +
                        func1(dp, i, ind-1, cuts) +
                        func1(dp, ind+1, j, cuts);
        if(mini > x)
            mini = x;
    }

    return dp[i][j] = mini;
}

////Recursion - > tabulation
int func2(vector<int> &cuts, int c)
{
    vector<vector<int>> dp(c+2, vector<int>(c+2, 0));
 
    for(int i = c; i >= 1; i--)
    {
        for(int j = 1; j <= c; j++)
        {
            if(i > j)
                continue;

            int mini = INT_MAX;
            for(int ind = i; ind <= j; ind++)
            {
                int x = cuts[j+1] - cuts[i-1] +
                                dp[i][ind-1] +
                                dp[ind+1][j];
                if(mini > x)
                    mini = x;
            }
            dp[i][j] = mini;
        }
    }

    return dp[1][c];
}
