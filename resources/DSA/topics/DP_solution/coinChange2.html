//Recursion
long func1(int *num, vector<vector<long>> &dp, int i, int target)
{
    if(i == 0)
    {
        return target%num[0]==0;
    }

    if(dp[i][target] != -1)
        return dp[i][target];

    long ways = func1(num, dp, i-1, target);

    if(num[i] <= target)
        ways +=func1(num, dp, i, target-num[i]);

    dp[i][target] = ways;
}

//Recursion -- > tabulation
long func2(int *num, int n, int target)
{
    vector<vector<long>> dp(n, vector<long>(target+1, 0));

    for(int tar = 0; tar <= target; tar++)
    {
        dp[0][tar] = (tar%num[0] == 0);
    }
    
    for(int i = 1; i < n; i++)
    {
        for(int tar = 0; tar <= target; tar++)
        {
            long way = dp[i-1][tar];
            if(num[i] <= tar)
                way += dp[i][tar-num[i]];

            dp[i][tar] = way;
        }
    }

    return dp[n-1][target];
}
