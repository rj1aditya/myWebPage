#include <bits/stdc++.h> 

//Recursion
int func1(vector<int>& a, vector<vector<int>> &dp, int i, int j)
{
	if(i > j)
		return 0;

	if(dp[i][j] != -1)	
		return dp[i][j];

	int maxi = INT_MIN;
	for(int ind = i; ind <= j; ind++)
	{
		int coinEarned = a[i-1] * a[ind] * a[j+1] +
										func1(a, dp, i, ind-1)+
										func1(a, dp, ind+1, j);
		maxi = max(maxi, coinEarned);
	}
	return dp[i][j] = maxi;
}

//Recursion-->Tabulation
int func2(vector<int>& a)
{
	int n = a.size()-2;
	vector<vector<int>> dp(n+2, vector<int>(n+2, 0));

	for(int i = n; i >= 1; i--)
	{
		for(int j = 1; j <= n; j++)
		{
			if(i > j)
				continue;

			int maxi = INT_MIN;
			for(int ind = i; ind <= j; ind++)
			{
				int coinEarned = a[i-1] * a[ind] * a[j+1] +
												dp[i][ind-1] +
												dp[ind+1][j];
				maxi = max(maxi, coinEarned);
			}
			dp[i][j] = maxi;
		}
	}

	return dp[1][n];
}
