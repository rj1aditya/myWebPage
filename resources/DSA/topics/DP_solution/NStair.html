Count Ways To Reach The N-th Stairs

C++ Solution


// C++ solution code goes here
#include <bits/stdc++.h>
int mod=1e9+7;
int fun(int n, vector<int> &dp)
{
    if(n <= 1)
      return 1;
    if(dp[n] != -1)
      return dp[n];
    return dp[n] = (fun(n-1, dp)%mod + fun(n-2,dp)%mod)%mod;
} 
int countDistinctWays(int n) {
    //  Write your code here.
  vector<int> dp(n+1, -1);
  return fun(n, dp);
}
