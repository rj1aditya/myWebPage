#include <bits/stdc++.h> 

//Recursion
int func1(vector<int> &arr, vector<vector<int>> &dp, int i, int j)
{
    if(i==j)
        return 0;

    if(dp[i][j] != -1)
        return dp[i][j];

    int mini = 1e9+7;
    for(int k = i; k < j; k++)
    {
        int steps = arr[i-1]*arr[k]*arr[j] + func1(arr, dp, i, k) + func1(arr, dp, k+1, j);
        if(steps < mini)
            dp[i][j] = mini = steps;
    }
    
    return dp[i][j];
}

//Recursion --> Tabulation
int func2(vector<int> &arr)
{
    int n = arr.size();
    vector<vector<int>> dp(n, vector<int>(n, 0));
    for(int i = 1; i < n; i++)
        dp[i][i] = 0;

    for(int i = n-1; i >= 1; i--)
    {
        for(int j = i+1; j < n; j++)
        {
            int mini = 1e9+7;
            for(int k = i; k < j; k++)
            {
                int steps = arr[i-1]*arr[k]*arr[j] + dp[i][k] + dp[k+1][j];
                if(steps < mini)
                    dp[i][j] = mini = steps;
            }
        }
    }

    return dp[1][n-1];
}
