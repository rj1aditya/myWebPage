// C++ solution code goes here
class Solution {
    public:
        //Brute Force
        string minWindow1(string s, string t) {
            int minIndex = -1;
            int minlen = 1e9;
    
            for(int i = 0; i < s.size(); i++)
            {
                map<char,int> hash;
                int cnt = 0;
                //Store all the t string characters with its freq into the hash
                for(int j = 0; j < t.size(); j++)
                {
                    hash[t[j]]++;
                }
    
                for(int k = i; k < s.size(); k++)
                {
                    if(hash[s[k]] > 0)
                        cnt++;
                    
                    hash[s[k]]--;
    
                    if(cnt == t.size())
                    {
                        if(k-i+1 < minlen)
                        {
                            minlen = k-i+1;
                            minIndex = i;
                        }
                        break;
                    }
                }
            }
            cout <<minIndex<<endl<<minlen;
            
            if(minIndex == -1)
                return "";
            return s.substr(minIndex, minlen);
        }
    
        string minWindow(string s, string t) {
            int l = 0, r = 0, cnt = 0, minIndex=-1;
            int minlen = 1e9;
            map<char,int> hash;
    
            int n = s.size();
            int m = t.size();
    
            //Store t string into the hash map
            for(int i = 0; i < m; i++) hash[t[i]]++;
    
            while(r < n)
            {
                if(hash[s[r]] > 0)
                    cnt++;
                hash[s[r]]--;
    
                while(cnt == m)
                {
                    if(minlen > r-l+1)
                    {
                        minlen = r-l+1;
                        minIndex = l;
                    }
                    hash[s[l]]++;
                    if(hash[s[l]] > 0)
                        cnt = cnt-1;
                    l++;
                }
                r++;
            }
            if(minIndex != -1)
                return s.substr(minIndex, minlen);
            
            return "";
        }
    
    };
