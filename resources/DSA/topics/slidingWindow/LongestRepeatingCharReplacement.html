Longest Repeating Character Replacement

C++ Solution


// C++ solution code goes here
class Solution {
    public:
        //Brute force
        int characterReplacement1(string s, int k) {
            int maxsubstr = 0;
            for(int i = 0; i < s.size(); i++)
            {
                vector<int> maxfreq(26, 0);
                int maxif = 0;
                for(int j = i; j < s.size(); j++)
                {
                    maxfreq[s[j] - 'A']++;
                    
                    maxif = max(maxif, maxfreq[s[j] - 'A']);
                    //Here we reduce the maxf from the len so that 
                    //we can know how many left string to change to make 
                    //whole substring same.
                    int len = j - i + 1 - maxif;
                    if(len <= k)
                    {
                        maxsubstr = max(maxsubstr, j - i + 1);
                    }
                    else
                        break;
                }
            }
            return maxsubstr;
        }
        int characterReplacement(string s, int k) {
            int maxsubstr = 0;
    
            int l = 0, r = 0, maxfreq = 0;
            vector<int> hashmap(26 , 0);
    
            while(r < s.size())
            {
                hashmap[s[r]-'A']++;
                maxfreq = max(maxfreq, hashmap[s[r] - 'A']);
    
                int len = r - l + 1 - maxfreq;
                if(len > k)
                {
                    hashmap[s[l]-'A']--;
                    maxfreq = 0;
                    //Finding new max freq from hashmap
                    for(int i = 0; i < 26; i++)
                        maxfreq = max(maxfreq, hashmap[i]);
    
                    l++;
                }
                else
                {
                    maxsubstr = max(maxsubstr, r - l + 1);
                }
                r++;
            }
            return maxsubstr;
        }
    };
