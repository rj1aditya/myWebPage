Binary Subarrays With Sum

C++ Solution


// C++ solution code goes here
class Solution {
    int numSubarrayWithS(vector<int>& nums, int goal)
    {
        if(goal < 0)
            return 0;

        int l = 0, r = 0, cnt = 0;
        int sum = 0;
        while(r < nums.size())
        {
            sum += nums[r];
            while(sum > goal)
            {
                sum = sum - nums[l];
                l++;
            }
            
            if(sum <= goal)
            {
                cnt = cnt + r - l + 1;
            }
            r++;
        }
        return cnt;
    }
public:
    int numSubarraysWithSum(vector<int>& nums, int goal) {
        int x1 = numSubarrayWithS(nums, goal-1);
        int x2 = numSubarrayWithS(nums, goal);

        return x2 - x1;
    }
};
