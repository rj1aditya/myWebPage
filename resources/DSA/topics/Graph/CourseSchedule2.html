class Solution
{
  public:
    vector<int> findOrder(int n, int m, vector<vector<int>> prerequisites) 
    {
        //code here
        int indegree[n] = {0};
        vector<int> adj[n];
        for(auto x: prerequisites)
            adj[x[1]].push_back(x[0]);
            
        
        //find indegree 0 node
        for(int i = 0; i < n; i++)
        {
            for(auto n: adj[i])
            {
                indegree[n]++;
            }
        }
        
        //take 0 indegree node in the queue
        queue<int> q;
        for(int i = 0; i < n; i++)
        {
            if(indegree[i] == 0)
                q.push(i);
        }
        
        vector<int> topo;
        while(!q.empty())
        {
            auto tmp = q.front();
            q.pop();
            topo.push_back(tmp);
            
            for(auto k: adj[tmp])
            {
                indegree[k]--;
                if(indegree[k] == 0)
                    q.push(k);
            }
            
        }
        if(topo.size() == n)
            return topo;
        
        return {};
        
    }
};
