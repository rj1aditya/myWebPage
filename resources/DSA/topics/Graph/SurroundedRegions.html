Surrounded Regions | Replace O's with X's

C++ Solution


// C++ solution code goes here
class Solution
{

    vector<int> dr{-1, 0, 1, 0};
    vector<int> dc{0, 1, 0, -1};
    int n1;
    int m1;
    void dfs(int row, int col, vector<vector<char>> &mat, vector<vector<int>> &vis)
    {
        vis[row][col] = 1;
        // ans[row][col] = '1';

        for (int i = 0; i < 4; i++)
        {
            int nr = row + dr[i];
            int nc = col + dc[i];
            if (nr >= 0 && nr < n1 && nc >= 0 && nc < m1 && vis[nr][nc] == 0 && mat[nr][nc] == 'O')
            {
                dfs(nr, nc, mat, vis);
            }
        }
    }

public:
    vector<vector<char>> fill(int n, int m, vector<vector<char>> mat)
    {
        // Set class member variables
        n1 = n;
        m1 = m;

        vector<vector<int>> vis(n, vector<int>(m, 0));

        // Make sure take M here Check 0th and m-1th row boundary
        for (int i = 0; i < m; i++)
        {
            if (vis[0][i] == 0 && mat[0][i] == 'O')
            {
                dfs(0, i, mat, vis);
            }

            if (vis[n - 1][i] == 0 && mat[n - 1][i] == 'O')
            {
                dfs(n - 1, i, mat, vis);
            }
        }

        // Make sure take N here Check 0th and n-1th col boundary
        for (int i = 0; i < n; i++)
        {
            if (vis[i][0] == 0 && mat[i][0] == 'O')
            {
                dfs(i, 0, mat, vis);
            }

            if (vis[i][m - 1] == 0 && mat[i][m - 1] == 'O')
            {
                dfs(i, m - 1, mat, vis);
            }
        }

        for (int i = 0; i < n; i++)
        {
            for (int j = 0; j < m; j++)
            {
                if (vis[i][j] == 0 && mat[i][j] == 'O')
                    mat[i][j] = 'X';
            }
        }

        return mat;
    }
};
