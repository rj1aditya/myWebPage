Dijkstra's Algorithm using set DS

C++ Solution


// C++ solution code goes here
#include <bits/stdc++.h> 
vector<int> dijkstra(vector<vector<int>> &vec, int vertices, int edges, 
int source) {
    // Write your code here.

    vector<pair<int, int>> adj[vertices];
    for(int i = 0; i < edges; i++)
    {
        int u = vec[i][0];
        int v = vec[i][1];
        int wt = vec[i][2];

        adj[u].push_back({v, wt});
        adj[v].push_back({u, wt});
    }

    set<pair<int,int>> st;
    st.insert({0, source});

    vector<int> path(vertices, INT_MAX);
    path[source] = 0;

    while(!st.empty())
    {
        auto it = st.begin();
        int node = it->second;
        int dis = it->first;
        st.erase(it);

        for(auto adjacentNode: adj[node])
        {
            int adjN = adjacentNode.first;
            int adjNWt = adjacentNode.second;

            if(dis+adjNWt < path[adjN])
            {
                if(path[adjN] != INT_MAX)
                {
                    st.erase({path[adjN], adjN});
                }
                path[adjN] = dis+adjNWt;
                st.insert({path[adjN], adjN});
            }
        }
    }
    return path;
}
