Rotten Oranges

C++ Solution


// C++ solution code goes here
class Solution {
    public:
        int orangesRotting(vector<vector<int>>& grid) {
            
            //pair --> {{row, col}, time}
            int n = grid.size();
            int m = grid[0].size();
            queue<pair<pair<int,int>, int>> q;
            //first visit the matrix and figure our initially how many orange are rotten & take those in a queue
            for(int i = 0; i < n; i++)
            {
                for(int j = 0; j < m; j++)
                {
                    if(grid[i][j] == 2)
                    {
                        q.push({{i,j}, 0});
                    }
                }
            }
    
            int time = 0;
            vector<vector<int>> vis(n, vector<int>(m, 0));
            //BFS Traversal
            while(!q.empty())
            {
                auto tmp = q.front();
                q.pop();
                int row = tmp.first.first;
                int col = tmp.first.second;
                int t = tmp.second;
    
                time = max(time, t);
    
                int dr[] = {-1, 0, 1, 0};
                int dc[] = {0, 1, 0, -1};
    
                //move in all 4 direction
                for(int i = 0; i < 4; i++)
                {
                    int R = row + dr[i];
                    int C = col + dc[i];
                    if(R >= 0 && R < n && C >= 0 && C < m && grid[R][C] == 1 && vis[R][C] != 1)
                    {
                        vis[R][C] = 1;
                        q.push({{R, C}, t+1});
                    }
                }
            }
    
            //Check all grid is rotten or not
            for(int i = 0; i < n; i++)
            {
                for(int j = 0; j < m; j++)
                {
                    if(grid[i][j] == 1 && vis[i][j] != 1)
                    {
                        return -1;
                    }
                }
            }
            return time;
        }
    };
