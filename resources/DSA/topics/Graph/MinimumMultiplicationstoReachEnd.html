Minimum Multiplications to Reach End

C++ Solution


// C++ solution code goes here
class Solution {
    int mod{100000};
    
  public:
    int minimumMultiplications(vector<int>& arr, int start, int end) {
        // code here
        
        //node, steps
        if(start == end)
            return 0;
        queue<pair<int, int>> q;
        q.push({start, 0});
        
        vector<int> steps(100000, 1e9);
        steps[start] = 0;
        
        while(!q.empty())
        {
            int nd = q.front().first;
            int stp = q.front().second;
            q.pop();
            
            for(auto it: arr)
            {
                int num = it*nd%mod;
                
                if(steps[num] > stp+1)
                {
                    if(num == end)
                        return stp+1;
                    steps[num]=stp+1;
                    q.push({num, steps[num]});
                }
            }
        }
        
        return -1;
        
    }
};
