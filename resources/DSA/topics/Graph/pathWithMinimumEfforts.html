Path With Minimum Effort

C++ Solution


// C++ solution code goes here
class Solution {
    public:
      int MinimumEffort(int rows, int columns, vector<vector<int>> &heights) {
          // code here
          
          using p = pair<int, pair<int,int>>;
          
          priority_queue<p, vector<p>, greater<p>> pq;
          pq.push({0, {0,0}});
          
          vector<vector<int>> path(rows, vector<int>(columns, 1e9));
          int dr[] = {-1, 0, 1, 0};
          int dc[] = {0, 1, 0, -1};
          
          while(!pq.empty())
          {
              auto it = pq.top();
              pq.pop();
              int dis = it.first;
              int r = it.second.first;
              int c = it.second.second;
              
              //we are taking ans here because we are using the priority queue
              //it will always keeps the less value distance at the top,
              //so if we reach the destination with any distance, means
              //that distance going to be the lesser one than any other one.
              if(r == rows-1 && c == columns-1)
                  return dis;
                  
              for(int i = 0; i < 4; i++)
              {
                  int newR = r + dr[i];
                  int newC = c + dc[i];
                  if(newR >=0 && newR < rows && newC >= 0 && newC < columns)
                  {
                      int newDist = max(abs(heights[r][c] - heights[newR][newC]), dis);
                      if(newDist < path[newR][newC])
                      {
                          pq.push({newDist, {newR, newC}});
                          path[newR][newC] = newDist;
                      }
                  }
              }
          }
          return -1;
      }
  };
