Shortest Path in Directed Acyclic Graph

C++ Solution


// C++ solution code goes here
class Solution {
    void dfs(int node, int vis[], vector<pair<int,int>> adj[], stack<int> &stk) {
        vis[node] = 1;
        for(auto it: adj[node]) {
            if(vis[it.first] == 0) {
                dfs(it.first, vis, adj, stk);
            }
        }
        stk.push(node);
    }
    
  public:
    vector<int> shortestPath(int N, int M, vector<vector<int>>& edges) {
        vector<pair<int,int>> adj[N];
        stack<int> stk;
        int vis[N] = {0};
        
        // Build adjacency list
        for(auto it: edges) {
            adj[it[0]].push_back({it[1], it[2]});
        }
        
        // Perform DFS to get topological sort
        for(int i = 0; i < N; i++) {
            if(vis[i] == 0) {
                dfs(i, vis, adj, stk);
            }
        }
        
        // Initialize distances
        vector<int> dist(N, INT_MAX);
        dist[0] = 0;  // Assuming 0 is the source node
        
        // Process nodes in topological order
        while(!stk.empty()) {
            int node = stk.top();
            stk.pop();
            
            if (dist[node] != INT_MAX) {
                for(auto it: adj[node]) {
                    int v = it.first;
                    int w = it.second;
                    if(dist[node] + w < dist[v]) {
                        dist[v] = dist[node] + w;
                    }
                }
            }
        }
        
        // Convert distances for non-reachable nodes to -1
        for(int i = 0; i < N; i++) {
            if(dist[i] == INT_MAX) {
                dist[i] = -1;
            }
        }
        
        return dist;
    }
};
