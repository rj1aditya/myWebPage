Number of Ways to Arrive at Destination

C++ Solution


// C++ solution code goes here
class Solution {
    
    public:
      int countPaths(int n, vector<vector<int>>& roads) {
          // code here
          int mod=(int)(1e9+7);
          vector<pair<long long,int>> adj[n];
          for(auto it: roads)
          {
              adj[it[0]].push_back({it[1], it[2]});
              adj[it[1]].push_back({it[0], it[2]});
          }
          
          priority_queue<pair<long long,int>, vector<pair<long long,int>>, greater<pair<long long,int>>> pq;
          pq.push({0, 0});
  
          vector<long long> dist(n, 1e18);
          vector<int> ways(n, 0);
          
          dist[0] = 0;
          ways[0] = 1;
          
          while(!pq.empty())
          {
              auto X = pq.top();
              pq.pop();
              int node = X.second;
              long long tm = X.first;
              
              for(auto it: adj[node])
              {
                  int adjN = it.first;
                  long long adjNTime = it.second;
                  
                  if(adjNTime + tm < dist[adjN])
                  {
                      dist[adjN] = adjNTime+tm;
                      pq.push({dist[adjN],adjN});
                      ways[adjN] = ways[node]%mod;
                  }
                  else if(adjNTime+tm == dist[adjN])
                  {
                      ways[adjN] = (ways[adjN] + ways[node])%mod;
                  }
              }
              
          }
          return ways[n-1]%mod;
      }
  };
