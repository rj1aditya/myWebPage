class comparator
{
    public:
    
    bool operator()(pair<int, int> &p1, pair<int, int> &p2)
    {
        return p1.second > p2.second;
    }
};

class Solution
{
	public:
	//Function to find the shortest distance of all the vertices
    //from the source vertex S.
    vector <int> dijkstra(int V, vector<vector<int>> adj[], int S)
    {
        // Code here
        priority_queue<pair<int, int>, vector<pair<int,int>>, comparator> pq;
        pq.push({0, S});
        
        vector<int> path(V, INT_MAX);
        path[S] = 0;
        
        while(!pq.empty())
        {
            auto tm = pq.top();
            pq.pop();
            
            int node = tm.second;
            int dis = tm.first;
            
            //Traverse all the adjacent nodes to node
            for(auto it: adj[node])
            {
                int edgeWt = it[1];
                int edgeNode = it[0];
                
                if(dis+edgeWt < path[edgeNode])
                {
                    path[edgeNode] = dis + edgeWt;
                    pq.push({path[edgeNode], edgeNode});
                }
            }
        }
        return path;
    }
};

class compare{

vector<int> dijkstra(vector<vector<int>> &vec, int vertices, int edges, int source) {
    // Write your code here.

    //adjacent graph
    vector<pair<int, int>> adj[vertices];
    for(int i = 0; i < edges; i++)
    {
        int u = vec[i][0];
        int v = vec[i][1];
        int wt = vec[i][2];

        adj[u].push_back({v, wt});
        adj[v].push_back({u, wt});
    }

    priority_queue< pair<int,int>, vector<pair<int,int>>, compare> pq;
    //{node, weight}
    pq.push({0, source});
    vector<int> path(vertices, INT_MAX);
    path[source] = 0;

    while(!pq.empty())
    {
        int node = pq.top().second;
        int dis = pq.top().first;

        pq.pop();
        for(auto adjacentN :adj[node])
        {
            int adN = adjacentN.first;
            int wt = adjacentN.second;
            if(dis+wt < path[adN])
            {
                path[adN] = dis+wt;
                pq.push({dis+wt, adN});
            }
        }

    }
    return path;
}
