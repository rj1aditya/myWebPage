// C++ solution code goes here
class Solution
{
	public:
	//Function to return list containing vertices in Topological order. 
	vector<int> topoSort(int V, vector<int> adj[]) 
	{
	    // code here
	    //Kanh's algo
	    
	    int indegree[V] = {0};
	    
	    //Figuring out the indegree of the nodes
	    for(int i = 0; i < V; i++)
	    {
	        for(auto x: adj[i])
	            indegree[x]++;
	    }
	    
	    queue<int> q;
	    
	    //Take those nodes in the queue first
	    //Whose indegree is 0
	    for(int i = 0; i < V; i++)
	    {
	        if(indegree[i] == 0)
	            q.push(i);
	    }
	    
	    vector<int> topo;
	    while(!q.empty())
	    {
	        auto tmp = q.front();
	        topo.push_back(tmp);
	        q.pop();
	        
	        for(auto n: adj[tmp])
	        {
	            //start removing the edge and reducing the degree
	            indegree[n]--;
	            //now after reducing the indegree, 
	            //if indegree become 0 then insert it in the queue
	            if(indegree[n] == 0)
	                q.push(n);
	        }
	    }
	    return topo;
	}
};
