Cheapest Flights Within K Stops

C++ Solution


// C++ solution code goes here
class Solution {
    public:
      int CheapestFLight(int n, vector<vector<int>>& flights, int src, int dst, int K)  {
          // Code here
          
          
          vector<pair<int, int>> adj[n];
          
          for(auto it: flights)
          {
              //0->1,100 means 0 to 1st node at 100 cost
              adj[it[0]].push_back({it[1], it[2]});
          }
          
          //{stops, {node, cost}}
          queue<pair<int, pair<int, int>>> q;
          q.push({0, {src, 0}});
          
          vector<int> desti(n+1, 1e9);
          desti[src] = 0;
          
          while(!q.empty())
          {
              auto it = q.front();
              q.pop();
              
              int node = it.second.first;
              int cost = it.second.second;
              int stops = it.first;
              
              if(stops > K)
                  continue;
              
              for(auto adjacentNodes: adj[node])
              {
                  int adjN = adjacentNodes.first;
                  int adjNCost = adjacentNodes.second;
                  
                  if(stops <= K && cost+adjNCost < desti[adjN] )
                  {
                      desti[adjN] = cost+adjNCost;
                      q.push({stops+1, {adjN, desti[adjN]}});
                  }
              }
          }
          if(desti[dst] == 1e9)
              return -1;
          return desti[dst];
      }
  };
