class Solution{
    public:
    string findOrder(string dict[], int N, int K) {
        //code here
        vector<int> adj[K];
        
        for(int i = 0; i < N-1; i++)
        {
            int len1 = dict[i].length();
            int len2 = dict[i+1].length();
            
            int x = 0;
            int y = 0;
            
            while(x < len1 && y < len2)
            {
                if(dict[i][x] != dict[i+1][y])
                {
                    adj[dict[i][x] - 'a'].push_back(dict[i+1][y]-'a');
                    break;
                }
                x++;
                y++;
            }
        }
        
        int indegree[K] = {0};
        
        queue<int> q;
        
        for(int i = 0; i < K; i++)
        {
            for(auto it: adj[i])
            {
                indegree[it]++;
            }
        }
        
        for(int i = 0; i < K; i++)
        {
            if(indegree[i] == 0)
                q.push(i);
        }
        
        vector<int> safeNode;
        while(!q.empty())
        {
            auto tmp = q.front();
            q.pop();
            safeNode.push_back(tmp);
            
            for(auto it: adj[tmp])
            {
                indegree[it]--;
                if(indegree[it] == 0)
                {
                    q.push(it);
                }
            }
        }
        
        string ans;
        for(int i = 0; i < safeNode.size(); i++)
            ans = ans + char(safeNode[i] + 'a');
        //cout << ans<<endl;
        return ans;
        
    }
};
