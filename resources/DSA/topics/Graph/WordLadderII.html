// C++ solution code goes here
class Solution {
    public:
        vector<vector<string>> findSequences(string beginWord, string endWord, vector<string>& wordList) {
            // code here
            
            //take wordList into the set DS
            unordered_set<string> st(wordList.begin(), wordList.end());
            
            //Create a Queue of string vector
            queue<vector<string>> q;
            
            //Take the beginWord into the queue
            q.push({beginWord});
            
            //Ans will be contain
            vector<vector<string>> ans;
            
            //Keep track those word which is used in the queue, So that later we can 
            //Remove that word from the set once the level is over
            int level = 0;
            vector<string> newWord;
            
            newWord.push_back(beginWord);
            
            while(!q.empty())
            {
                //q.front will return the vector of string
                vector<string> lis = q.front();
                q.pop();
                
                
                
                //Check if level is over, so we can remove words from the set
                if(lis.size() > level)
                {
                    level++;
                    for(auto it: newWord)
                        st.erase(it);
                    
                    newWord.clear();
                }
                
                //take the last word only fron the stack
                string word = lis.back();
                
                //Take the ans if last word in list matches the endWord
                if(word == endWord)
                {
                    if(ans.size() == 0)
                        ans.push_back(lis);
                    else
                    {
                        int size = ans[0].size();
                        if(lis.size() == size)
                            ans.push_back(lis);
                    }
                }
                
                for(int i = 0; i < word.size(); i++)
                {
                    string orignalWord = word;
                    for(char c='a'; c <= 'z'; c++)
                    {
                        word[i] = c;
                        if(st.find(word) != st.end())
                        {
                            lis.push_back(word);
                            q.push(lis);
                            newWord.push_back(word);
                            lis.pop_back();
                        }
                    }
                    word = orignalWord;
                }
            }
            
            return ans;
            
        }
    };
