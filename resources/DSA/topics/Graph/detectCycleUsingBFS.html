Detect a Cycle in an Undirected Graph using BFS

// C++ solution code goes here
class Solution {
    bool detectLoopDFS(int N, int parent, vector<int> adj[], int vis[])
    {
        vis[N] = 1;
        
        for(auto adjacent: adj[N])
        {
            if(vis[adjacent] == 0)
            {
                if(detectLoopDFS(adjacent, N, adj, vis) == true)
                    return true;
            }
            else if(adjacent != parent)
                return true;
        }
        
        return false;
    }

    bool detectLoopBFS(int N, int V, vector<int> adj[], int vis[])
    {
        queue<pair<int,int>> q;
        q.push({N, -1});

        vis[N] = 1;
        
        while(!q.empty())
        {
            auto tmp = q.front();
            q.pop();
            int node = tmp.first;
            int parent = tmp.second;
            
            for(auto adjacent: adj[node])
            {
                if(!vis[adjacent])
                {
                    vis[adjacent] = 1;
                    q.push({adjacent, node});
                }
                else if(parent != adjacent)
                    return true;
            }
        }
        
        return false;
    }
  public:
    // Function to detect cycle in an undirected graph.
    bool isCycle(int V, vector<int> adj[]) {
        // Code here
        int vis[V] = {0};
        
        for(int i = 0; i < V; i++)
        {
            if(vis[i] == 0)
            {
                //if(detectLoopBFS(i, V, adj, vis) == true)
                    //return true;
                if(detectLoopDFS(i, -1, adj, vis) == true)
                    return true;
            }
        }
        return false;
    }
};
