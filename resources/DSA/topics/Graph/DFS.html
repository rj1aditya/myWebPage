Depth-First Search (DFS)

C++ Solution


// C++ solution code goes here
class Solution {
    void dfs(int &V, vector<int> adj[], int vis[], int node, vector<int> &ans)
    {
        ans.push_back(node);
        vis[node] = 1;
        
        for(auto &nod :adj[node])
        {
            //not visited
            if(!vis[nod])
            {
                dfs(V, adj, vis, nod, ans);
            }
        }
    }

  public:
    // Function to return a list containing the DFS traversal of the graph.
    vector<int> dfsOfGraph(int V, vector<int> adj[]) {
        // Code here
        int vis[V] = {0};
        vector<int> ans;
        
        dfs(V, adj, vis, 0, ans);
        return ans;
    }
};
