class Solution {
    bool detectLoopDFS(int N, int parent, vector<int> adj[], int vis[], int pathVisited[])
    {
        vis[N] = 1;
        pathVisited[N] = 1;

        for(auto adjacent: adj[N])
        {
            if(vis[adjacent] == 0)
            {
                if(detectLoopDFS(adjacent, N, adj, vis, pathVisited) == true)
                    return true;
            }
            else if(pathVisited[adjacent])
                return true;
        }
        pathVisited[N] = 0;
        return false;
    }

  public:
    // Function to detect cycle in an undirected graph.
    bool isCyclic(int V, vector<int> adj[]) {
        // Code here
        int vis[V] = {0};
        int pathVisited[V] = {0};
        
        for(int i = 0; i < V; i++)
        {
            if(vis[i] == 0)
            {
                //BFS won't work here
                //if(detectLoopBFS(i, V, adj, vis, pathVisited) == true)
                    //return true;
                if(detectLoopDFS(i, -1, adj, vis, pathVisited) == true)
                    //return true;
            }
        }
        return false;
    }
};
