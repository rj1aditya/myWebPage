class Solution1 {
    //BFS    
        vector<int> color;
    public:
    bool checkBipartite(vector<vector<int>>& graph, int curr)
    {
        queue<int> q;
        q.push(curr); 
        color[curr] = 1;
        while(!q.empty())
        {
            int node = q.front();
            q.pop();
            
            for(auto i: graph[node])
            {
                if(color[i] == -1)
                {
                    color[i] = 1 - color[node];
                    q.push(i);
                }
                else if(color[i] == color[node])
                {
                    return false;
                }
            }
            
            
        }
        return true;
    }
    bool isBipartite(vector<vector<int>>& graph) {
        color = vector<int>(graph.size(), -1);
        
        //Thought: I think no meaning of doing if, to check component wise
        //Coz if single component is not Bipartite then we are returning 
        //false, not checking other graph
        for(int i = 0; i < graph.size(); i++)
        {
            if(color[i] == -1)
            {
                if(!checkBipartite(graph, i))
                    return false;
            }
        }
        return true;
    }
};
    
    class Solution
    {
        //DFS
    public:
    vector<int> vis, col;
    bool dfs(vector<vector<int>> &graph, int curr)
    {
        if(col[curr] == -1)
            col[curr] = 1;

        for (int node : graph[curr])
        {
            if (col[node] == -1)
            {
                col[node] = 1 - col[curr];
                if (dfs(graph, node) == false)
                    return false;
            }
            else
            {
                if (col[node] == col[curr])
                    return false;
            }
        }
        return true;
    }
    
    bool isBipartite(vector<vector<int>> &graph)
    {
        int n = graph.size();
        col = vector<int>(n,-1);

        for (int i = 0; i < n; ++i)
        {
            if (col[i] == -1)
            {
                if(dfs(graph, i) == false)
                    return false;
            }
        }

        return true;
    }
};
